结论

  • 指针慢，但是不会慢很多，单点100ms左右。
  • 大部分情况下，替罪羊快。
  • PBDS效率还挺高的，不卡常可以直接用。

template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;
constexpr int N(1e6 + 5), S(2e8);


std::mt19937 rng(123);

int rnd(int l, int r) {
  if (l > r) std::swap(l, r);
  return l + rng() % (r - l + 1);
}

char buf[S], *pBuf = buf + S;

struct Treap {
  struct Node {
    unsigned key;
    int val;
    Node *ch[2], *p;

    int siz;
    Node(int x) : key(rng()), val(x), ch{0, 0}, p(0), siz(1) {}
    
    void *operator new(size_t n) { return pBuf -= n; }
    void pushup() {
      siz = 1;
      if (ch[0]) siz += ch[0]->siz;
      if (ch[1]) siz += ch[1]->siz;
    }
    void sch(int d, Node *x) {
      if (ch[d] = x) x->p = this;
    }
    int dir() {
      return this == p->ch[1];
    }
    void rotate() {
      Node *f = p;
      int k = dir();
      p = f->p;
      if (p) p->ch[f->dir()] = this;
      f->sch(k, ch[!k]), sch(!k, f);
      f->pushup();
    }
  } *root;
  Treap() : root(0) {}
  void insert(int x) {
    Node *p = new Node(x);
    if (!root) {
      root = p;
    } else {
      Node *o = root;
      for (;;) {
        o->siz++;
        int k = x > o->val;
        if (o->ch[k]) {
          o = o->ch[k];
        } else {
          o->sch(k, p);
          break;
        }
      }
    }
    while (p->p && p->key < p->p->key) p->rotate();
    p->pushup();
    if (!p->p) root = p;
  
  }
  int order_of_key(int x) {
    auto o = root;
    int ans = 0;
    while (o) {
      if (x > o->val) {
        ans += (o->ch[0] ? o->ch[0]->siz : 0) + 1;
        o = o->ch[1];
      } else {
        o = o->ch[0];
      }
    }
    return ans;
  }
};

struct SegTree {
  struct Node {
    Node *ls, *rs;
    int w;
    Node() : ls(0), rs(0), w(0) {}
    void *operator new(size_t n) { return pBuf -= n; }
  };
  void ins(Node *&o, int l, int r, int x) {
    if (!o) o = new Node;
    o->w++;
    if (l == r) return;
    int m = l + r >> 1;
    if (x <= m) {
      ins(o->ls, l, m, x);
    } else {
      ins(o->rs, m + 1, r, x);
    }
  }
  int ask(Node *o, int l, int r, int x) {
    if (!o || x <= l) return 0;
    if (r < x) return o->w;
    int m = l + r >> 1;
    return ask(o->ls, l, m, x) + ask(o->rs, m + 1, r, x);
  }

  Node *root;
  SegTree() : root(0) {}
  void insert(int x) {
    ins(root, 0, 1e6, x);
  }
  int order_of_key(int x) {
    return ask(root, 0, 1e6, x);
  }
};

namespace Treap2 {
struct TreapNode {
  unsigned key;
  int ls, rs, w, siz;
} t[2000005];
void pushup(int o) {
  t[o].siz = t[t[o].ls].siz + t[t[o].rs].siz + 1;
}
int merge(int x, int y) {
  if (!x) return y;
  if (!y) return x;
  return t[x].key < t[y].key ? (t[x].rs = merge(t[x].rs, y), pushup(x), x) : (t[y].ls = merge(x, t[y].ls), pushup(y), y);
}
void split(int o, int k, int &x, int &y) {
  if (!o) {
    x = y = 0;
    return;
  }
  if (t[o].w < k) {
    x = o;
    split(t[o].rs, k, t[o].rs, y);
  } else {
    y = o;
    split(t[o].ls, k, x, t[o].ls);
  }
  pushup(o);
}
int rank(int o, int x) {
  int ans = 0;
  while (o) {
    if (x > t[o].w) {
      ans += t[t[o].ls].siz + 1;
      o = t[o].rs;
    } else {
      o = t[o].ls;
    }
  }
  return ans;
}
struct Treap2 {
  int root;
  Treap2() : root(0) {}
  void insert(int x) {
    t[++cnt] = {(unsigned)rng(), 0, 0, x, 1};
    int a, b;
    split(root, x, a, b);
    root = merge(merge(a, cnt), b);
  }
  int order_of_key(int x) {
    return rank(root, x);
  }
};

    if (r < x) return t[o].w;

    return ask(t[o].ls, l, m, x) + ask(t[o].rs, m + 1, r, x);
  }
  void insert(int x) {
    ins(root, 0, 1e6, x);
  }
  int order_of_key(int x) {
    return ask(root, 0, 1e6, x);
  }
};

template <class T>
std::pair<double, int> test(unsigned seed) {
  pBuf = std::end(buf);
  cnt = 0;
  double sta = (double)clock() / CLOCKS_PER_SEC;
  std::mt19937 rng(seed);
  const int n = 20;
  std::vector<T> t(n);
  int ans = 0;
  std::vector<int> a(1e6);
  std::iota(a.begin(), a.end(), 0);
  std::reverse(a.begin(), a.end());
  std::shuffle(a.begin(), a.begin() + a.size()    , rng);
  for (int i = 0; i < 1e6; i++) {                    
    int x = rng() % n;
    if (rng() & 1) {
      t[x].insert(a[i]);
    } else {
      ans ^= t[x].order_of_key(a[i]);
    }
  }                                                  
  return {(double)clock() / CLOCKS_PER_SEC - sta, ans};
}


namespace Treap3 {
struct Node {
  unsigned key;
  int p, ch[2], val, siz;
} t[2000005];
void pushup(int o) {
  t[o].siz = t[t[o].ch[0]].siz + t[t[o].ch[1]].siz + 1;
}
void sch(int o, int d, int x) {
  t[o].ch[d] = x, t[x].p = o;
}
int dir(int o) {
  return o == t[t[o].p].ch[1];
}
void rotate(int o) {
  int p = t[o].p, k = dir(o);
  t[o].p = t[p].p;
  if (t[p].p) t[t[p].p].ch[dir(p)] = o;
  sch(p, k, t[o].ch[!k]), sch(o, !k, p);
  pushup(p);
}

struct Treap3 {
  int root;
  Treap3() : root(0) {}
  void insert(int x) {
    int p = ++cnt;
    t[p] = {(unsigned)rng(), 0, {0, 0}, x, 1};
    if (!root) {
      root = p;
    } else {
      int o = root;
      for (;;) {
        int k = x > t[o].val;
        t[o].siz++;
        if (!t[o].ch[k]) {
          sch(o, k, p);
          break;
        } else {
          o = t[o].ch[k];
        }
      }
    }
    while (t[p].p && t[p].key < t[t[p].p].key) rotate(p);
    pushup(p);
    if (!t[p].p) root = p;
  }
  int order_of_key(int x) {
    int o = root, ans = 0;
    while (o) {
      if (t[o].val < x) {
        ans += t[t[o].ch[0]].siz + 1;
        o = t[o].ch[1];
      } else {
        o = t[o].ch[0];
      }
    }
    return ans;
  }
};
}

namespace SCT {
#define ls ch[0]
#define rs ch[1]
struct Node {
  int ch[2], val, siz;
} t[N];
void pushup(int o) { t[o].siz = t[t[o].ls].siz + t[t[o].rs].siz + 1; }
bool bad(int o) {
  return t[o].siz * 0.73 < std::max(t[t[o].ls].siz, t[t[o].rs].siz);
}
int a[N], top;
void dfs(int x) {
  if (!x) return;
  dfs(t[x].ls), a[++top] = x, dfs(t[x].rs);
}
int build(int l, int r) {
  if (l > r) return 0;
  int m = l + r >> 1, o = a[m];
  t[o].ls = build(l, m - 1), t[o].rs = build(m + 1, r);
  pushup(o);
  return o;
}

struct SCT {
  int root;
  SCT() : root(0) {}
  void insert(int x) {
    int p = ++cnt;
    t[p] = {{0, 0}, x, 1};
    if (!root) {
      root = p;
    } else {
      int o = root;
      top = 0;
      for (;;) {
        a[++top] = o;
        t[o].siz++;
        int k = x > t[o].val;
        if (t[o].ch[k]) {
          o = t[o].ch[k];
        } else {
          t[o].ch[k] = p;
          break;
        }
      }
      for (int i = 1; i <= top; i++) {
        if (bad(a[i])) {
          int &o = i > 1 ? t[a[i - 1]].ch[a[i] == t[a[i - 1]].ch[1]] : root;
          top = 0;
          dfs(o);
          o = build(1, top);
          break;
        }
      }
    }
  }
  int order_of_key(int x) {
    int o = root, ans = 0;
    while (o) {
      if (t[o].val < x) {
        ans += t[t[o].ls].siz + 1;
        o = t[o].rs;
      } else {
        o = t[o].ls;
      }
    }
    return ans;
  }
};

}
#include <ext/pb_ds/assoc_container.hpp>
template <class T, class U = __gnu_pbds::null_type, class Comp = std::less<T>>
using Tree = __gnu_pbds::tree<T, U, Comp, __gnu_pbds::rb_tree_tag, __gnu_pbds::tree_order_statistics_node_update>;

void solve() {
  std::vector<unsigned> seeds(3);
  for (int i = 0; i < 3; i++) {
    seeds[i] = rng();
  }
  double t;
  int ans;
#define TEST(T)                                                         \
  t = 0, ans = 0;                                                       \
  for (auto s : seeds) { auto [u, v] = test<T>(s); t += u, ans ^= v; } \
  std::cout << #T << " " << t << " " << ans << "\n";

  TEST(Treap);
  TEST(Treap2::Treap2);
  TEST(Treap3::Treap3);
  TEST(SCT::SCT);
  TEST(SegTree);
  TEST(SegTree2);
  TEST(Tree<int>);

}

}
