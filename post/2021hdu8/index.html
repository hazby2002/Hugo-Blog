using LL = long long;
using PII = std::pair<int, int>;

LL pw[20];
int m;
LL cal(LL x) {
  if (!x) return 1;
  
  static int a[20];
  static LL f[20], dp[20][500];
  int c = 0;
  for (; x; x /= 10) a[c++] = x % 10;
  std::reverse(a, a + c);
  memset(f, -1, sizeof f);
  std::function<LL(int, bool, bool)> dfs = [&](int p, bool lim, bool lead) -> LL {
    if (p == c) return 1;
    if (!lim && !lead && ~f[p]) return f[p];
    LL ans = 0;
    int up = lim ? a[p] : 9;
    for (int i = 0; i <= up; i++) {
      if (i == 7) continue;
      ans += dfs(p + 1, lim && i == up, lead && i == 0);
    }
    if (!lim && !lead) f[p] = ans;
    return ans;
  };
  LL ans = dfs(0, true, true);
  for (int i = 1; i < m; i++) {
    for (int j = 0; j < m; j++) {
      dp[c][j] = j == i;
    }
    for (int j = c - 1; j >= 0; j--) {
      for (int k = 0; k < m; k++) {
        dp[j][k] = 0;
        if (!k || k * pw[c - j] % m == i) continue;
        for (int x = 0; x < 10; x++) {
          if (x == 7) continue;
          dp[j][k] += dp[j + 1][(k * 10 + x) % m];
        }
      }
    }
    // memset(dp, -1, sizeof dp);
    std::function<LL(int, int, bool, bool)> dfs = [&](int p, int r, bool lim, bool lead) -> LL {
      if (p == c) return r == i;
      if (!lead && (!r || r * pw[c - p] % m == i)) return 0;
      if (!lim && !lead && ~dp[p][r]) return dp[p][r];
      LL ans = 0;
      int up = lim ? a[p] : 9;
      for (int i = 0; i <= up; i++) {
        if (i == 7) continue;
        ans += dfs(p + 1, (r * 10 + i) % m, lim && i == up, lead && i == 0);
      }
      if (!lim && !lead) dp[p][r] = ans;
      return ans;
    };
    ans -= dfs(0, 0, true, true);
  }
  return ans;
}

Buying Snacks

\[\begin{aligned} A &= 1 + x + x^2\\ B &= x + 2x^2 + x^3\\ F_n &= AF_{n-1} + BF_{n-2}\\ \lambda^2-A\lambda - B = 0 &\Rightarrow \lambda = \frac{A \pm \sqrt{A^2+4B}}{2}\\ \sqrt{A^2+4B} &= 1 + 3x + x^2\\ \lambda_1 = 1+2x+x^2,& \lambda_2=-x\\ F_n &= P(1+2x+x^2)^n+Q(-x)^n\\ \begin{cases} F_0 &= 1\\ F_1 &= 1 + x + x ^2\\ \end{cases} &\Rightarrow \begin{cases} P &= \frac{1+2x+x^2}{1 + 3x + x^2} \\ Q &= \frac{x}{1 + 3x + x^2}\\ \end{cases}\\ F_n &= \frac{(1+x)^{2n+2}+(-1)^nx^{n+1}}{1 + 3x + x^2} \end{aligned}\]

using LL = long long;
using PII = std::pair<int, int>;
constexpr int P(998244353);
inline void inc(int &x, int y) {
  x += y;
  if (x >= P) x -= P;
}
inline int mod(LL x) {
  return x % P;
}
int fpow(int x, int k = P - 2) {
  int r = 1;
  for (; k; k >>= 1, x = 1LL * x * x % P) {
    if (k & 1) r = 1LL * r * x % P;
  }
  return r;
}
constexpr int N(1e6 + 5);
int a[N], pw[N];
int sum[N << 2], b[N << 2], tag[N << 2], max[N << 2];
#define ls o << 1
#define rs o << 1 | 1
int sumMod(int x, int y) {
  return x + y >= P ? x + y - P : x + y;
}
void pushup(int o) {
  sum[o] = sumMod(sum[ls], sum[rs]);
  b[o] = sumMod(b[ls], b[rs]);
  max[o] = std::max(max[ls], max[rs]);
}
void modify(int o, int x) {
  tag[o] += x;
  sum[o] = (sum[o] + 1LL * (pw[x] - 1) * b[o]) % P;
  b[o] = 1LL * b[o] * pw[x] % P;
}
void pushdown(int o) {
  if (tag[o]) {
    modify(ls, tag[o]);
    modify(rs, tag[o]);
    tag[o] = 0;
  }
}
void update1(int o, int l, int r, int x, int y) {
  if (max[o] == 0) return;
  if (l == r) {
    max[o]--;
    if (max[o]) {
      int x = a[l] & -a[l];
      inc(sum[o], P - x);
      a[l] -= x;
    } else {
      sum[o] = 0;
      b[o] = 0;
      tag[o] = 0;
      a[l] = 0;
    }
    return;
  }
  pushdown(o);
  int m = l + r >> 1;
  if (x <= m) update1(ls, l, m, x, y);
  if (y > m) update1(rs, m + 1, r, x, y);
  pushup(o);
}
void update2(int o, int l, int r, int x, int y) {
  if (x <= l && r <= y) {
    modify(o, 1);
    return;
  }
  pushdown(o);
  int m = l + r >> 1;
  if (x <= m) update2(ls, l, m, x, y);
  if (y > m) update2(rs, m + 1, r, x, y);
  pushup(o);
}
int ask(int o, int l, int r, int x, int y) {
  if (r < x || y < l) return 0;
  if (x <= l && r <= y) return sum[o];
  pushdown(o);
  int m = l + r >> 1;
  return sumMod(ask(ls, l, m, x, y), ask(rs, m + 1, r, x, y));
}
void build(int o, int l, int r) {
  tag[o] = 0;
  if (l == r) {
    if (a[l]) {
      sum[o] = a[l] % P;
      max[o] = __builtin_popcount(a[l]);
      b[o] = (1 << std::__lg(a[l])) % P;
    } else {
      sum[o] = max[o] = b[o] = 0;
    }
    
    return;
  }
  int m = l + r >> 1;
  build(ls, l, m), build(rs, m + 1, r);
  pushup(o);
}

void solve() {
  int n;
  std::cin >> n;
  for (int i = 1; i <= n; i++) {
    std::cin >> a[i];
  }
  build(1, 1, n);
  int q;
  std::cin >> q;
  while (q--) {
    int opt, l, r;
    std::cin >> opt >> l >> r;
    if (opt == 1) {
      std::cout << ask(1, 1, n, l, r) << "\n";
    } else if (opt == 2) {
      update1(1, 1, n, l, r);
    } else {
      update2(1, 1, n, l, r);
    }
  }
}

  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  pw[0] = 1;
  for (int i = 1; i <= 1e5; i++) pw[i] = 2LL * pw[i - 1] % P;
  int t;
  std::cin >> t;

    solve();
  }

void dft(int *a, int n) {
  assert((n & n - 1) == 0);
  for (int k = n >> 1; k; k >>= 1) {
    for (int i = 0; i < n; i += k << 1) {
      for (int j = 0; j < k; j++) {
        int y = a[i + j + k];
        a[i + j + k] = 1LL * (a[i + j] - y + P) * w[k + j] % P;
        inc(a[i + j], y);
      }
    }
  }
}
void idft(int *a, int n) {
  assert((n & n - 1) == 0);
  for (int k = 1; k < n; k <<= 1) {
    for (int i = 0; i < n; i += k << 1) {
      for (int j = 0; j < k; j++) {
        int x = a[i + j], y = 1LL * a[i + j + k] * w[k + j] % P;
        a[i + j + k] = x - y < 0 ? x - y + P : x - y;
        inc(a[i + j], y);
      }
    }
  }
  for (int i = 0, inv = P - (P - 1) / n; i < n; i++)
    a[i] = 1LL * a[i] * inv % P;
  std::reverse(a + 1, a + n);
}
void dft(Poly &a) { dft(a.data(), a.size()); }

  if (a.size() <= 8 || b.size() <= 8) {

    for (int i = 1; i < n; i++) inc(f[i], f[i - 1]);
    for (int i = 2; i < n; i++) {
      ans = (ans + 1LL * val[i] * f[i - 2] - 1LL * pw[i] * g[i - 2]) % P;
    }
  }
  std::cout << (ans % P + P) % P << "\n";
}

using LL = long long;

}
struct Reader {

  Reader &operator>>(T &w) {
    char c, p = 0;
    for (; !std::isdigit(c = gc());) if (c == '-') p = 1;

struct Node {
  Node *ls, *rs;

} t[N * 30], *cur;
int n, k, q, cnt;
void update(Node *&o, int l, int r, int x, int y, int z) {
  if (o == t) {
    o = cur++;
    *o = {t, t, 0, 0, 0};
  }
  if (x <= l && r <= y) {

}
int cal(Node *o, int l, int r, int x, int y) {
  Node *p = o->ls, *q = o->rs;
  int m = l + r >> 1, c = p->add, d = q->add;
  while (m - l > 1) {
    p = p->rs;
    c += p->add;
    l = l + m >> 1;
  }
  while (r - m > 1) {
    q = q->ls;
    d += q->add;
    r = r + m >> 1;
  }

  while (r - l > 1) {
    m = l + r >> 1;
    if (m <= x || m < y && std::max(a, c + p->rs->val) >= std::max(b, d + q->ls->val)) {
      if (x <= m && m < y) smin(ans, std::max({a, c + p->rs->val, b, d + q->ls->val}));
      l = m;   
      smax(b, d + q->ls->val);
      p = p->rs, q = q->rs;
      c += p->add, d += q->add;
    } else {     
      if (x <= m && m < y) smin(ans, std::max({a, c + p->rs->val, b, d + q->ls->val}));
      r = m;
      smax(a, c + p->rs->val);
      p = p->ls, q = q->ls;
      c += p->add, d += q->add;
    }
  }
  return ans;
}
void add(Node *&o, int l, int r, int x, int y, int z) { // [l * k, r * k), [x, y)
  if (o == t) {
    o = cur++;
    *o = {t, t, 0, 0, 0};
  }
  if (x <= l * k && r * k <= y) {
    o->val += z;
    o->add += z;
    return;
  }
  if (r - l == 1) {
    update(o, 0, k, std::max(0, x - l * k), std::min(k, y - l * k), z);
    return;
  }
  int m = l + r >> 1;
  if (x < m * k) add(o->ls, l, m, x, y, z);
  if (y > m * k) add(o->rs, m, r, x, y, z);
  if (x <= (m - 1) * k && y >= (m + 1) * k) {
    o->mid += z;
  } else if (m == x / k || m == y / k || m - 1 == x / k || m - 1 == y / k) {
    o->mid = cal(o, l, r, 0, k);
  }
  o->val = std::min({o->ls->val, o->rs->val, o->mid}) + o->add;
}
int ask(Node *o, int l, int r, int x, int y) {
  if (x <= l * k && r * k <= y) return o->val;
  int m = l + r >> 1, ans = 1e9;
  if (x <= (m - 1) * k) smin(ans, ask(o->ls, l, m, x, y));
  if (y >= (m + 1) * k) smin(ans, ask(o->rs, m, r, x, y));
  if (x <= (m - 1) * k && y >= (m + 1) * k) {
    smin(ans, o->mid);
  } else if (x < m * k && m * k < y) {
    smin(ans, cal(o, l, r, std::max(0, x - (m - 1) * k), std::min(k, y - m * k + 1)));
  }
  return ans + o->add;
}

void solve() {
  cur = t + 1;
  cin >> n >> k >> q;
  int m = (n - 1) / k + 1;
  Node *root = t;
  while (q--) {
    int opt, l, r, x;
    cin >> opt >> l >> r;
    l--;
    if (opt == 1) {
      cin >> x;
      add(root, 0, m, l, r, x);
    } else {
      std::cout << ask(root, 0, m, l, r) << "\n";
    }
  }
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  *t = {t, t, 0, 0, 0};
  int t;
  cin >> t;
  while (t--) {
    solve();
  }
  return 0;
}
