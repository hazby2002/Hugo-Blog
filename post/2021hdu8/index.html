Buying Snacks

\[\begin{aligned} A &= 1 + x + x^2\\ B &= x + 2x^2 + x^3\\ F_n &= AF_{n-1} + BF_{n-2}\\ \lambda^2-A\lambda - B = 0 &\Rightarrow \lambda = \frac{A \pm \sqrt{A^2+4B}}{2}\\ \sqrt{A^2+4B} &= 1 + 3x + x^2\\ \lambda_1 = 1+2x+x^2,& \lambda_2=-x\\ F_n &= P(1+2x+x^2)^n+Q(-x)^n\\ \begin{cases} F_0 &= 1\\ F_1 &= 1 + x + x ^2\\ \end{cases} &\Rightarrow \begin{cases} P &= \frac{1+2x+x^2}{1 + 3x + x^2} \\ Q &= \frac{x}{1 + 3x + x^2}\\ \end{cases}\\ F_n &= \frac{(1+x)^{2n+2}+(-1)^nx^{n+1}}{1 + 3x + x^2} \end{aligned}\]

#include <bits/stdc++.h>
constexpr int P(998244353);
inline void inc(int &x, int y) {
  x += y;
  if (x >= P) x -= P;
}
constexpr int N(4e4 + 5);
int a[N], inv[N];
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  inv[1] = 1;
  for (int i = 2; i <= 5e4; i++) inv[i] = 1LL * (P - P / i) * inv[P % i] % P;
  int t;
  std::cin >> t;
  while (t--) {
    int n, m;
    std::cin >> n >> m;
    a[0] = 1;
    for (int i = 1, k = 2 * n + 2; i <= m; i++) {
      if (i <= k) {
        a[i] = 1LL * a[i - 1] * inv[i] % P * (k - i + 1) % P;
      } else {
        a[i] = 0;
      }
    }
    if (n + 1 <= m) inc(a[n + 1], n & 1 ? P - 1 : 1);
    for (int i = 0; i <= m; i++) {
      a[i + 1] = (a[i + 1] + (P - 3LL) * a[i]) % P;
      inc(a[i + 2], P - a[i]);
    }
    for (int i = 1; i <= m; i++) {
      std::cout << a[i] << " ";
    }
    std::cout << "\n";
  }
  return 0;
}

  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  std::cin >> t;
  while (t--) {
    int n;
    std::cin >> n;
    std::vector<PII> a(n);
    for (int i = 0; i < n; i++) {
      std::cin >> a[i].first >> a[i].second;
    }
    std::vector<int> vis(n);
    vis[0] = 1;
#define double LL
#define sqrt(x) (x)
    std::vector<double> min(n, 2e18);
    auto sqr = [&](double x) { return x * x; };
    for (int i = 1; i < n; i++) {
      min[i] = sqrt(sqr(a[i].first - a[0].first) + sqr(a[i].second - a[0].second));
    }
    double ans = 0;
    for (int i = 0; i < n - 1; i++) {
      double mi = 2e18;
      int x = -1;
      for (int j = 0; j < n; j++) {
        if (!vis[j] && smin(mi, min[j])) x = j;
      }
      assert(~x);
      smax(ans, min[x]);
      vis[x] = 1;
      for (int j = 0; j < n; j++) {
        if (!vis[j]) smin(min[j], sqrt(sqr(a[x].first - a[j].first) + sqr(a[x].second - a[j].second)));
      }
    }
    std::cout << ans << "\n";
  }
  
  return 0;
}

// Author:  HolyK
// Created: Thu Aug 12 12:32:44 2021
#include <bits/stdc++.h>
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;
constexpr int P(998244353);
inline void inc(int &x, int y) {
  x += y;
  if (x >= P) x -= P;
}
inline int mod(LL x) {
  return x % P;
}
int fpow(int x, int k = P - 2) {
  int r = 1;
  for (; k; k >>= 1, x = 1LL * x * x % P) {
    if (k & 1) r = 1LL * r * x % P;
  }
  return r;
}
constexpr int N(1e6 + 5);
int a[N], pw[N];
int sum[N << 2], b[N << 2], tag[N << 2], max[N << 2];
#define ls o << 1
#define rs o << 1 | 1
int sumMod(int x, int y) {
  return x + y >= P ? x + y - P : x + y;
}
void pushup(int o) {
  sum[o] = sumMod(sum[ls], sum[rs]);
  b[o] = sumMod(b[ls], b[rs]);
  max[o] = std::max(max[ls], max[rs]);
}
void modify(int o, int x) {
  tag[o] += x;
  sum[o] = (sum[o] + 1LL * (pw[x] - 1) * b[o]) % P;
  b[o] = 1LL * b[o] * pw[x] % P;
}
void pushdown(int o) {
  if (tag[o]) {
    modify(ls, tag[o]);
    modify(rs, tag[o]);
    tag[o] = 0;
  }
}
void update1(int o, int l, int r, int x, int y) {
  if (max[o] == 0) return;
  if (l == r) {
    max[o]--;
    if (max[o]) {
      int x = a[l] & -a[l];
      inc(sum[o], P - x);
      a[l] -= x;
    } else {
      sum[o] = 0;
      b[o] = 0;
      tag[o] = 0;
      a[l] = 0;
    }
    return;
  }
  pushdown(o);
  int m = l + r >> 1;
  if (x <= m) update1(ls, l, m, x, y);
  if (y > m) update1(rs, m + 1, r, x, y);
  pushup(o);
}
void update2(int o, int l, int r, int x, int y) {
  if (x <= l && r <= y) {
    modify(o, 1);
    return;
  }
  pushdown(o);
  int m = l + r >> 1;
  if (x <= m) update2(ls, l, m, x, y);
  if (y > m) update2(rs, m + 1, r, x, y);
  pushup(o);
}
int ask(int o, int l, int r, int x, int y) {
  if (r < x || y < l) return 0;
  if (x <= l && r <= y) return sum[o];
  pushdown(o);
  int m = l + r >> 1;
  return sumMod(ask(ls, l, m, x, y), ask(rs, m + 1, r, x, y));
}
void build(int o, int l, int r) {
  tag[o] = 0;
  if (l == r) {
    if (a[l]) {
      sum[o] = a[l] % P;
      max[o] = __builtin_popcount(a[l]);
      b[o] = (1 << std::__lg(a[l])) % P;
    } else {
      sum[o] = max[o] = b[o] = 0;
    }
    
    return;
  }
  int m = l + r >> 1;
  build(ls, l, m), build(rs, m + 1, r);
  pushup(o);
}

void solve() {
  int n;
  std::cin >> n;
  for (int i = 1; i <= n; i++) {
    std::cin >> a[i];
  }
  build(1, 1, n);
  int q;
  std::cin >> q;
  while (q--) {
    int opt, l, r;
    std::cin >> opt >> l >> r;
    if (opt == 1) {
      std::cout << ask(1, 1, n, l, r) << "\n";
    } else if (opt == 2) {
      update1(1, 1, n, l, r);
    } else {
      update2(1, 1, n, l, r);
    }
  }
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  pw[0] = 1;
  for (int i = 1; i <= 1e5; i++) pw[i] = 2LL * pw[i - 1] % P;
  int t;
  std::cin >> t;
  while (t--) {
    solve();
  }
  return 0;
}

// Author:  HolyK
// Created: Thu Aug 12 13:41:13 2021
#include <bits/stdc++.h>
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;

namespace Polynomial {
using Poly = std::vector<int>;
constexpr int P(998244353), G(3);
inline void inc(int &x, int y) { (x += y) >= P ? x -= P : 0; }
inline int mod(int64_t x) { return x % P; }
inline int fpow(int x, int k = P - 2) {
  int r = 1;
  for (; k; k >>= 1, x = 1LL * x * x % P)
    if (k & 1) r = 1LL * r * x % P;
  return r;
}
template <int N>
std::array<int, N> getOmega() {
  std::array<int, N> w;
  for (int i = N >> 1, x = fpow(G, (P - 1) / N); i; i >>= 1, x = 1LL * x * x % P) {
    w[i] = 1;
    for (int j = 1; j < i; j++) w[i + j] = 1LL * w[i + j - 1] * x % P;
  }
  return w;
}
auto w = getOmega<1 << 22>();
Poly &operator*=(Poly &a, int b) { for (auto &x : a) x = 1LL * x * b % P; return a; }
Poly operator*(Poly a, int b) { return a *= b; }
Poly operator*(int a, Poly b) { return b * a; }
Poly &operator/=(Poly &a, int b) { return a *= fpow(b); }
Poly operator/(Poly a, int b) { return a /= b; }
Poly &operator+=(Poly &a, Poly b) {
  a.resize(std::max(a.size(), b.size()));
  for (int i = 0; i < b.size(); i++) inc(a[i], b[i]);
  return a;
}
Poly operator+(Poly a, Poly b) { return a += b; }
Poly &operator-=(Poly &a, Poly b) {
  a.resize(std::max(a.size(), b.size()));
  for (int i = 0; i < b.size(); i++) inc(a[i], P - b[i]);
  return a;
}
Poly operator-(Poly a, Poly b) { return a -= b; }
Poly operator-(Poly a) { for (auto &x : a) x ? x = P - x : 0; return a; }
Poly &operator>>=(Poly &a, int x) {
  if (x >= (int)a.size()) {
    a.clear();
  } else {
    a.erase(a.begin(), a.begin() + x);
  }
  return a;
}
Poly &operator<<=(Poly &a, int x) {
  a.insert(a.begin(), x, 0);
  return a;
}
Poly operator>>(Poly a, int x) { return a >>= x; }
Poly operator<<(Poly a, int x) { return a <<= x; }
inline Poly &dotEq(Poly &a, Poly b) {
  assert(a.size() == b.size());
  for (int i = 0; i < a.size(); i++) a[i] = 1LL * a[i] * b[i] % P;
  return a;
}
inline Poly dot(Poly a, Poly b) { return dotEq(a, b); }
void norm(Poly &a) {
  if (!a.empty()) {
    a.resize(1 << std::__lg(a.size() * 2 - 1));
  }
}
void dft(int *a, int n) {
  assert((n & n - 1) == 0);
  for (int k = n >> 1; k; k >>= 1) {
    for (int i = 0; i < n; i += k << 1) {
      for (int j = 0; j < k; j++) {
        int y = a[i + j + k];
        a[i + j + k] = 1LL * (a[i + j] - y + P) * w[k + j] % P;
        inc(a[i + j], y);
      }
    }
  }
}
void idft(int *a, int n) {
  assert((n & n - 1) == 0);
  for (int k = 1; k < n; k <<= 1) {
    for (int i = 0; i < n; i += k << 1) {
      for (int j = 0; j < k; j++) {
        int x = a[i + j], y = 1LL * a[i + j + k] * w[k + j] % P;
        a[i + j + k] = x - y < 0 ? x - y + P : x - y;
        inc(a[i + j], y);
      }
    }
  }
  for (int i = 0, inv = P - (P - 1) / n; i < n; i++)
    a[i] = 1LL * a[i] * inv % P;
  std::reverse(a + 1, a + n);
}
void dft(Poly &a) { dft(a.data(), a.size()); }
void idft(Poly &a) { idft(a.data(), a.size()); }
Poly operator*(Poly a, Poly b) {
  int len = a.size() + b.size() - 1;
  if (a.size() <= 8 || b.size() <= 8) {
    Poly c(len);
    for (size_t i = 0; i < a.size(); i++)
      for (size_t j = 0; j < b.size(); j++)
        c[i + j] = (c[i + j] + 1LL * a[i] * b[j]) % P;
    return c;
  }
  int n = 1 << std::__lg(len - 1) + 1;
  a.resize(n), b.resize(n);
  dft(a), dft(b);
  dotEq(a, b);
  idft(a);
  a.resize(len);
  return a;
}
}
using namespace Polynomial;
constexpr int N(2e6 + 5);
int fac[N], ifac[N], pw[N], ipw[N];
void init(int n) {
  pw[0] = ipw[0] = 1;
  for (int i = 1; i <= n; i++) pw[i] = pw[i - 1] * 10LL % P;
  ipw[1] = fpow(10);
  for (int i = 2; i <= n; i++) ipw[i] = 1LL * ipw[i - 1] * ipw[1] % P;
  fac[0] = ifac[0] = 1;
  for (int i = 1; i <= n; i++) fac[i] = 1LL * fac[i - 1] * i % P;
  ifac[n] = fpow(fac[n]);
  for (int i = n - 1; i; i--) ifac[i] = 1LL * ifac[i + 1] * (i + 1) % P;
}
int bin(int n, int m) {
  if (m < 0 || m > n) return 0;
  return 1LL * fac[n] * ifac[m] % P * ifac[n - m] % P;
}
int val[N], sum[N];
void solve() {
  int k;
  std::string s;
  std::cin >> k >> s;
  int n = s.length();
  val[0] = 0;
  for (int i = 1; i <= n; i++) {
    val[i] = (val[i - 1] * 10LL + s[i - 1] - '0') % P;
  }
  int ans = val[n];
  if (k >= 2) {
    int s1 = 1;
    for (int i = 1; i < n; i++) {
      ans = mod(ans + (val[n] - 1LL * val[i] * pw[n - i] + val[n - i]) % P * s1);
      s1 = (s1 * 2LL - bin(i - 1, k - 2)) % P;
    }
  }
  if (k >= 3) {
    Poly f(n), g(n);
    f[n - 3] = 1;
    for (int i = 1; i < n - 2; i++) f[n - 3 - i] = (f[n - 2 - i] * 2LL - bin(i - 1, k - 3) + P) % P;
    for (int i = 1; i <= n; i++) g[i - 1] = 1LL * val[i] * ipw[i] % P;
    g = f * g;
    for (int i = 1; i < n; i++) inc(f[i], f[i - 1]);
    for (int i = 2; i < n; i++) {
      ans = (ans + 1LL * val[i] * f[i - 2] - 1LL * pw[i] * g[i - 2]) % P;
    }
  }
  std::cout << (ans % P + P) % P << "\n";
}
int main() {
  // freopen("t.in", "r", stdin);
  // freopen("t.out", "w", stdout);
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  init(1e6);
  int t;
  std::cin >> t;
  while (t--) {
    solve();
  }
  return 0;
}
