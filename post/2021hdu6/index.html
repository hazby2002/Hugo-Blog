Might and Magic

凸函数的和仍是凸函数。

// Author:  HolyK
// Created: Thu Aug  5 20:56:29 2021
#include <bits/stdc++.h>
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  std::cin >> t;
  while (t--) {
    int cp, cm, h, a, d, n;
    std::cin >> cp >> cm >> h >> a >> d >> n;
    h = (h + cp - 1) / cp;
    LL ans = 0;
    for (int i = 1, j; i <= a; i = j + 1) {
      int t = (h - 1) / i;
      j = t ? (h - 1) / t : 1e9;
      t++;
      int l = a - j, r = a - i;
      smax(l, 0);
      smin(r, n);
      if (l > r) continue;
      smax(ans, 1LL * cp * std::max(1, n - l - d) * t);
      auto f = [&](LL k) {
        return cm * k * (n - l - k) + cp * (t - k);
      };
      r = std::min(t, n - l);
      smax(ans, f(r));
      if (1 <= r) smax(ans, f(1));
      LL x = (1LL * (n - l) * cm - cp) / (2 * cm);
      if (x <= r && x >= 0) smax(ans, f(x));
      x++;
      if (x <= r && x >= 0) smax(ans, f(x));
    }
    std::cout << ans << "\n";
  }
  return 0;
}

0 tree

在不改变 \(\sum w_e\) 的情况下把边权和点权都集中到一条边。

#include <bits/stdc++.h>

constexpr int N(1e4 + 5);
using LL = long long;
using PII = std::pair<int, int>;
std::vector<PII> g[N];
int a[N];
bool ok;
struct Info {
  int x, y;
  LL z;
};
std::vector<Info> ans; 
int f[N], d[N];
LL val[N];

void dfs(int x) {
  for (auto &v : g[x]) {
    int y = v.first;
    if (y == f[x]) continue;
    val[y] = v.second;
    f[y] = x;
    d[y] = d[x] + 1;
    dfs(y);
  }
  if (d[x] > 1) {
    if (a[x]) {
      ans.push_back({x, f[f[x]], a[x]});
      a[f[f[x]]] ^= a[x];
      val[x] += a[x];
      val[f[x]] -= a[x];
      a[x] = 0;
    }
    if (val[x] & 1) {
      ok = false;
      return;
    }
    if (val[x] < 0) {
      ans.push_back({x, f[f[x]], -val[x] / 2});
      ans.push_back({x, f[f[x]], -val[x] / 2});
    } else if (val[x] > 0) {
      ans.push_back({f[f[x]], x, val[x] / 2});
      ans.push_back({f[f[x]], x, val[x] / 2});
    }
    val[f[x]] += val[x];
    val[x] = 0;
  }
}
void solve() {
  int n;
  std::cin >> n;
  for (int i = 1; i <= n; i++) {
    std::cin >> a[i];
    g[i].clear();
  }
  for (int i = 1, x, y, z; i < n; i++) {
    std::cin >> x >> y >> z;
    g[x].push_back({y, z});
    g[y].push_back({x, z});
  }
  if (n == 1) {
    std::cout << (a[1] == 0 ? "YES\n0\n" : "NO\n");
    return;
  }
  ok = true;
  ans.clear();
  for (int i = 1; i <= n; i++) {
    if (g[i].size() != 1) continue;
    d[i] = 0;
    f[i] = 0;
    val[i] = 0;
    dfs(i);    
    int j = g[i][0].first;
    if (a[i]) ans.push_back({i, j, a[i]});
    val[j] += a[i];
    if (val[j] > 0 || val[j] & 1 || a[i] != a[j]) {
      ok = false;
      break;
    }
    if (val[j]) ans.push_back({i, j, -val[j] / 2}), ans.push_back({i, j, -val[j] / 2});
    break;
  }
  if (ok) {
    std::cout << "YES\n";
    std::cout << ans.size() << "\n";
    for (auto &v : ans) {
      std::cout << v.x << " " << v.y << " " << v.z << "\n";
    }
  } else {
    std::cout << "NO\n";
  }
}
int main() {
  int t;
  std::cin >> t;
  while (t--) {
    solve();
  }
  return 0;
}
