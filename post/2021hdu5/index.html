Miserable Faith

LCT 裸题。

#include <bits/stdc++.h>
using LL = long long;
constexpr int N(1e5 + 5);
std::vector<int> g[N];
struct Node {
  int ch[2], fa, siz, top;
} t[N];
int n, m, in[N], out[N], cnt, dep[N], f[18][N];
int lca(int x, int y) {
  if (dep[x] < dep[y]) std::swap(x, y);
  for (int i = 0; dep[x] != dep[y]; i++) {
    if (dep[x] - dep[y] >> i & 1) {
      x = f[i][x];
    }
  }
  if (x == y) return x;
  for (int i = 17; i >= 0; i--) {
    if (f[i][x] != f[i][y]) {
      x = f[i][x];
      y = f[i][y];
    }
  }
  return f[0][x];
}
class FenwickTree {
 public:
  void add(int l, int r, LL z) {
    add(v, l, z), add(v, r + 1, -z);
    add(vi, l, 1LL * l * z), add(vi, r + 1, -1LL * (r + 1) * z);
  }
  LL ask(int l, int r) {
    return ask(v, r) * (r + 1) - ask(v, l - 1) * l - ask(vi, r) + ask(vi, l - 1);
  }
  void clear() {
    for (int i = 1; i <= n; i++) {
      v[i] = vi[i] = 0;
    }
  }
 private:
  LL v[N], vi[N];
  void add(LL c[], int p, LL x) {
    for (; p <= n; p += p & -p) c[p] += x;
  }
  LL ask(LL c[], int p) {
    LL r = 0;
    for (; p; p -= p & -p) r += c[p];
    return r;
  }
} fen;
void update(int x, int y) {
  fen.add(in[x], out[x], y);
}
int ask(int x) {
  return fen.ask(in[x], in[x]);
}
void dfs(int x, int p) {
  t[x].ch[0] = t[x].ch[1] = 0;
  t[x].fa = p;
  t[x].siz = 1;
  t[x].top = x;
  
  in[x] = ++cnt;
  f[0][x] = p;
  for (int y : g[x]) {
    if (y == p) continue;
    dep[y] = dep[x] + 1;
    dfs(y, x);
  }
  out[x] = cnt;
  if (p) update(x, 1);
}
void pushup(int o) {
  t[o].siz = 1 + t[t[o].ch[0]].siz + t[t[o].ch[1]].siz;
  t[o].top = t[o].ch[0] ? t[t[o].ch[0]].top : o;
}
void sch(int o, int d, int x) {
  t[o].ch[d] = x;
  t[x].fa = o;
}
bool nroot(int o) {
  return t[t[o].fa].ch[0] == o || t[t[o].fa].ch[1] == o;
}
void rotate(int o) {
  int f = t[o].fa, ff = t[f].fa, k = o == t[f].ch[1];
  if (nroot(f)) {
    t[ff].ch[f == t[ff].ch[1]] = o;
  }
  t[o].fa = ff;
  sch(f, k, t[o].ch[!k]);
  sch(o, !k, f);
  pushup(f);
  pushup(o);
}

void splay(int o) {
  for (; nroot(o); rotate(o)) {
    int f = t[o].fa;
    if (nroot(f)) {
      rotate((o == t[f].ch[1]) == (f == t[t[f].fa].ch[1]) ? f : o);
    }
  }
}
LL ans;

void access(int o) {
  for (int x = 0; o; o = t[x = o].fa) {
    splay(o);
    ans += 1LL * (1 + t[t[o].ch[0]].siz) * (t[x].siz - t[t[o].ch[1]].siz);
    if (t[o].ch[1]) update(t[t[o].ch[1]].top, 1);
    if (x) update(t[x].top, -1);
    t[o].ch[1] = x;
    
    pushup(o);
  }
}

void solve() {
  std::cin >> n >> m;
  for (int i = 1; i <= n; i++) {
    g[i].clear();
  }
  for (int i = 1, x, y; i < n; i++) {
    std::cin >> x >> y;
    g[x].push_back(y);
    g[y].push_back(x);
  }
  fen.clear();
  ans = 0;
  cnt = 0;
  dfs(1, 0);
  for (int i = 1; i < 18; i++) {
    for (int j = 1; j <= n; j++) {
      f[i][j] = f[i - 1][f[i - 1][j]];
    }
  }
  while (m--) {
    int opt, x, y;
    std::cin >> opt;
    if (opt == 4) {
      std::cout << ans << "\n";
    } else if (opt == 1) {
      std::cin >> x >> y;
      access(x);
    } else if (opt == 2) {
      std::cin >> x >> y;
      int z = lca(x, y);
      std::cout << ask(x) + ask(y) - 2 * ask(z) << "\n";
    } else {
      std::cin >> x;
      std::cout << fen.ask(in[x], out[x]) - 1LL * ask(x) * (out[x] - in[x] + 1) << "\n";
    }
  }
}
int main() {
  std::ios::sync_with_stdio(0);
  std::cin.tie(0);
  int t;
  std::cin >> t;
  while (t--) {
    solve();
  }
  return 0;
}

String Mod

由单位根反演 \[ [n | x] = \frac 1 n \sum_{k=0}^{n-1}\omega^{kx} \] 得 \[ \begin{aligned} ans_{i, j} &= \sum_{x,y}\binom{L}{x, y, L-x-y}(k-2)^{L-x-y} [n|x-i][n|y-j]\\ &= \frac1{n^2}\sum_{x,y}\binom{L}{x, y, L-x-y}(k-2)^{L-x-y}\sum_{p=0}^{n-1}\omega^{p(x-i)}\sum_{q=0}^{n-1}\omega^{q(y-j)}\\ &= \frac1{n^2}\sum_{p=0}^{n-1}\sum_{q=0}^{n-1}\omega^{-pi}\omega^{-qj}\sum_{x,y}\binom{L}{x, y, L-x-y}(k-2)^{L-x-y}\omega^{px}\omega^{qy}\\ &= \frac1{n^2}\sum_{p=0}^{n-1}\sum_{q=0}^{n-1}\omega^{-pi}\omega^{-qj}(k - 2 + \omega^p+\omega^q)^L \end{aligned} \] 令 \[ \begin{aligned} A_{i, p} &= \omega^{-ip},\\ B_{p, q} &= (k - 2 + \omega^p+\omega^q)^L,\\ C_{q, j} &= \omega^{-qj}\\ \end{aligned} \] 则 \[ ans = A \times B \times C \]

复杂度 \(O(n^3 + n^2\log L)\)。

int fpow(int x, int k = P - 2) {
  int r = 1;
  for (; k; k >>= 1, x = 1LL * x * x % P) {
    if (k & 1) r = 1LL * r * x % P;
  }
  return r;
}

    LL l;

    l %= P - 1;
    std::vector<int> w(n);
    w[0] = 1;
    w[1] = fpow(13, (P - 1) / n);
    for (int i = 2; i < n; i++) w[i] = 1LL * w[i - 1] * w[1] % P;
    using Matrix = std::vector<std::vector<int>>;    
    auto mul = [&](Matrix a, Matrix b) {
      Matrix c(n, std::vector<int>(n));
      for (int k = 0; k < n; k++) {
         for (int i = 0; i < n; i++) {
           for (int j = 0; j < n; j++) {
             c[i][j] = mod(c[i][j] + 1LL * a[i][k] * b[k][j]);
           }
         }
      }
      return c;
    };
    Matrix a(n, std::vector<int>(n)), b = a;
    for (int i = 0; i < n; i++) {
      for (int j = 0; j < n; j++) {
        a[i][j] = w[(n - i * j % n) % n];
        b[i][j] = fpow((k - 2LL + w[i] + w[j]) % P, l);
      }
    }
    a = mul(mul(a, b), a);
    int in = fpow(n);
    in = mod(1LL * in * in);
    for (int i = 0; i < n; i++) {
      for (int j = 0; j < n; j++) {

Array

给定序列 \(a_n\)，求有严格众数的区间数。

\(n \leq 10^6, 0 \leq a_i \leq 10^6\)。

原题 Yazid 的新生舞会。

考虑计算每种数作为严格众数的区间数。

设当前考虑的数为 \(x\)，那么令 \(b_i = \begin{cases} 1, & a_i = x \\ -1, & a_i \neq x \end{cases}\)。

一个区间存在严格众数，当且仅当 \(\sum_{i = l}^rb_i > 0\)。

对 \(b_i\) 做前缀和 \(s_i\)，转化为 \(s_i > s_j\)。

考虑将 \(b_i\) 划分成一些区间 \([l, r)\) 满足 \(b_l = 1, b_{l+1} = b_{l + 2} = \dots = b_{r-1} = -1, b_r = 1\)。

设按顺序访问区间 \([l, r)\) 时，会依次进行下面的操作：

  1. \(\operatorname{Query}(s_l), \operatorname{Query}(s_{l+1}), \dots, \operatorname{Query}(s_{r-1})\).
  2. \(\operatorname{Add}(s_l, 1), \operatorname{Add}(s_{l+1}, 1), \dots, \operatorname{Add}(s_{r-1}, 1)\).

其中 \(\operatorname{Add}(pos, val)\) 表示在 \(pos\) 处加 \(val\)，\(\operatorname{Query}(pos)\) 表示查询 \((-\infty, pos)\) 的和。

区间加等差数列，区间求和

考虑将单点加前缀求和转化成后缀加单点查询，那么操作 1 变成区间 \([s_{r-1}, s_l)\) 加 \(1, 2, 3, 4, \dots\)，区间 \([s_l, \infty)\) 加 \(r - l\)，操作 2 变成区间求和。

这个可以用线段树维护区间加等差数列，也可以用树状数组维护三阶前缀和。

复杂度 \(O(n\log n)\)。

考场上写的线段树因常数太大没有通过，如果写树状数组应该可以通过。

using LL = long long;
constexpr int N(2e6 + 5);
LL sum[N << 2], t1[N << 2], t2[N << 2];
int tag[N << 2];
#define ls o << 1
#define rs o << 1 | 1
void add(int o, int l, int r, LL z1, LL z2) {
  t1[o] += z1, t2[o] += z2;
  sum[o] += (r - l + 1) * z1 + (r - l + 1) * (r - l + 2) / 2 * z2;
}
void pushdown(int o, int l, int r) {
  if (tag[o]) {
    tag[ls] = tag[rs] = 1;
    sum[ls] = sum[rs] = t1[ls] = t1[rs] = t2[ls] = t2[rs] = 0;
    tag[o] = 0;
  }
  if (t1[o] || t2[o]) {
    int m = l + r >> 1;
    add(ls, l, m, t1[o], t2[o]);
    add(rs, m + 1, r, t1[o] + t2[o] * (m + 1 - l), t2[o]);
    t1[o] = t2[o] = 0;
  }
}
void pushup(int o) {
  sum[o] = sum[ls] + sum[rs];
}
void update(int o, int l, int r, int x, int y, LL z1, LL z2) {
  if (x <= l && r <= y) {
    add(o, l, r, z1, z2);
    return;
  }
  pushdown(o, l, r);
  int m = l + r >> 1;
  if (y <= m) {
    update(ls, l, m, x, y, z1, z2);
  } else if (x > m) {
    update(rs, m + 1, r, x, y, z1, z2);
  } else {
    update(ls, l, m, x, m, z1, z2);
    update(rs, m + 1, r, m + 1, y, z1 + (m + 1 - x) * z2, z2);
  }
  pushup(o);
}
LL ask(int o, int l, int r, int x, int y) {
  if (r < x || y < l) return 0;
  if (x <= l && r <= y) return sum[o];
  pushdown(o, l, r);
  int m = l + r >> 1;
  return ask(ls, l, m, x, y) + ask(rs, m + 1, r, x, y);
}
std::pair<int, int> a[N];
#define ALL 1, -n, n
void solve() {
  int n;
  cin >> n;
  for (int i = 1; i <= n; i++) {
    cin >> a[i].first;
    a[i].second = i;
  }
  std::sort(a + 1, a + 1 + n);
  LL ans = 0;
  for (int i = 1, j; i <= n; i = j) {
    for (j = i; j <= n && a[i].first == a[j].first; j++);
    tag[1] = 1, sum[1] = t1[1] = t2[1] = 0;
    // for (int i = 1; i <= n * 4; i++) sum[i] = t1[i] = t2[i] = 0;
    // std::vector<LL> tmp(2 * n + 1);
    for (int k = i; k < j; k++) {
      int r = (k - i) * 2 - (k == i ? 0 : a[k - 1].second), l = (k - i) * 2 - a[k].second + 1;
      update(ALL, l, r, 0, 1);
      update(ALL, r + 1, n, r - l + 1, 0);
      // for (int i = l; i <= r; i++) tmp[i + n] += i - l + 1, std::cerr << tmp[i + n] << " " << ask(1, -n, n, i, i) << "\n";
      // for (int i = r + 1; i <= n; i++) tmp[i + n] += r - l + 1, std::cerr << tmp[i + n] << " " << ask(1, -n, n, i, i) << "\n";
      r = (k - i + 1) * 2 - a[k].second - 1, l = (k - i + 1) * 2 - (k + 1 == j ? n + 1 : a[k + 1].second);
      ans += ask(ALL, l, r);
      // for (int i = l; i <= r; i++) ans += tmp[i + n];
    }
  }
  std::cout << ans << "\n";
}

区间加，区间求和

当 \(b\) 的前缀和 \(s_i\) 是 \(s_0, s_1, \dots, s_i\) 中的最小值时，\(i\) 作为右端点显然不会对答案有贡献，如果后面有连续的一段 \(-1\)，它们作为右端点也不会有贡献，所以这段区间的操作 1可以跳过，只需要操作 2，在这里相当于区间加 1。

设两个相邻的满足 \(s\) 是前缀最小值的位置依次为 \(i, j\)，必然有 \(b_j = -1, s_{j - 1} - s_{i} = 0\)，即 \(1\) 的个数和 \(-1\) 的个数相同。所以既需要查询也需要修改的位置个数为1的个数的两倍，这些位置可以直接暴力进行操作 1 和操作 2。

用树状数组维护区间加区间求和，复杂度 \(O(n \log n)\) 可以通过。

}

}
struct Reader {
  template <class T>
  Reader &operator>>(T &w) {

    for (; !std::isdigit(c = gc());) if (c == '-') p = 1;
    for (w = c & 15; std::isdigit(c = gc()); w = w * 10 + (c & 15)) ;

constexpr int N(1e6 + 5);
int n, a[N];
std::vector<int> g[N];
LL c[N << 1][2];
int t[N << 1], cnt;
void norm(int p) {
  if (t[p] != cnt) t[p] = cnt, c[p][0] = c[p][1] = 0;
}
void add(int p, int x) {
  int i = p;
  for (p += n + 1; p <= n * 2 + 1; p += p & -p) {
    norm(p);
    c[p][0] += x;
    c[p][1] += i * x;
  }
}
LL ask(int p) {
  LL r = 0;
  int i = p + 1;
  for (p += n + 1; p; p -= p & -p) {
    norm(p);
    r += c[p][0] * i - c[p][1];
  }
  return r;
}
void add(int l, int r, int x) {
  add(l, x), add(r + 1, -x);
}
LL ans;
void calc(std::vector<int> p) {
  int j = 1, min = 0, sum = 0;
  cnt++;
  add(0, 0, 1);
  for (int i : p) {
    while (j < i && sum > min + 1) {
      j++;
      ans += ask(--sum - 1);
      add(sum, sum, 1);
    }
    if (j < i) {
      add(sum - (i - j), sum - 1, 1);
      sum -= i - j;
      j = i;
    }

    smin(min, sum);
    j++;
    ans += ask(++sum - 1);
    add(sum, sum, 1);
  }
  while (j <= n && sum > min + 1) {
    j++;
    ans += ask(--sum - 1);
    add(sum, sum, 1);
  }
}

void solve() {
  cin >> n;
  for (int i = 1; i <= n; i++) {
    cin >> a[i];
    g[a[i]].push_back(i);
  }
  ans = 0;
  for (int i = 1; i <= n; i++) {
    if (g[a[i]].empty()) continue;
    calc(g[a[i]]);
    g[a[i]].clear();
  }
  std::cout << ans << "\n";
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  cin >> t;
  while (t--) {
    solve();
  }
  return 0;
}

线性算法

基于上述用树状数组维护区间加区间求和的方法，考虑到每次查询的区间端点只会加减一，或者直接跳过一大段 \(-1\)（满足前缀最小值的情况，此时答案直接清零了）。把树状数组改成数组，直接暴力移动指针即可。

复杂度 \(O(n)\)。

    for (; !std::isdigit(c = gc());) if (c == '-') p = 1;
    for (w = c & 15; std::isdigit(c = gc()); w = w * 10 + (c & 15)) ;
    if (p) w = -w;
    return *this;
  }
} cin;

constexpr int N(1e6 + 5);
int n, a[N];
std::vector<int> g[N];
LL c[N << 1][2], s[2];
int t[N << 1], cnt;
void norm(int p) {
  if (t[p] != cnt) t[p] = cnt, c[p][0] = c[p][1] = 0;
}
void add(int p, int x) {
  int i = p;
  p += n + 1;
  norm(p);
  c[p][0] += x;
  c[p][1] += x * i;
}
void add(int l, int r, int x) {
  add(l, x), add(r + 1, -x);
}
LL ans;
void calc(std::vector<int> p) {
  int j = 1, min = 0, sum = 0;
  cnt++;
  add(0, 0, 1);
  s[0] = s[1] = 0;
  for (int i : p) {
    while (j < i && sum > min + 1) {
      j++;
      int p = --sum + n + 1;
      norm(p);
      s[0] -= c[p][0];
      s[1] -= c[p][1];
      ans += sum * s[0] - s[1];
      add(sum, sum, 1);
    }
    if (j < i) {
      add(sum - (i - j), sum - 1, 1);
      sum -= i - j;
      j = i;
      s[0] = s[1] = 0;
    }
    smin(min, sum);
    j++;
    int p = sum + n + 1;
    norm(p);
    s[0] += c[p][0];
    s[1] += c[p][1];
    ans += ++sum * s[0] - s[1];
    add(sum, sum, 1);
  }
  while (j <= n && sum > min + 1) {
    j++;
    int p = --sum + n + 1;
    norm(p);
    s[0] -= c[p][0];
    s[1] -= c[p][1];
    ans += sum * s[0] - s[1];
  }
}

void solve() {
  cin >> n;
  for (int i = 1; i <= n; i++) {
    cin >> a[i];
    g[a[i]].push_back(i);
  }
  ans = 0;
  for (int i = 1; i <= n; i++) {
    if (g[a[i]].empty()) continue;
    calc(g[a[i]]);
    g[a[i]].clear();
  }
  std::cout << ans << "\n";
}

inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;

template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;

constexpr int N(1e5 + 5);
int n, a[N];
std::vector<PII> g[N];
LL f[N], diam;
void dfs(int x, int p) {
  f[x] = 0;
  for (auto &v : g[x]) {
    int y = v.first;
    int z = v.second;
    if (y == p) continue;
    dfs(y, x);
    smax(diam, f[x] + f[y] + z);
    smax(f[x], f[y] + z);
  }
}
LL h[N][2], limit;
bool dfs2(int x, int p) {
  h[x][0] = h[x][1] = 1e15;
  std::pair<LL, int> max[3];
  for (int i = 0; i < g[x].size(); i++) {
    int y = g[x][i].first;
    int z = g[x][i].second;
    if (y == p) continue;
    if (!dfs2(y, x)) return false;
    std::pair<LL, int> len(std::min(h[y][0] + std::max(0, z - a[y]), h[y][1] + z), i);
    if (len >= max[0]) {
      max[2] = max[1];
      max[1] = max[0];
      max[0] = len;
    } else if (len >= max[1]) {
      max[2] = max[1];
      max[1] = len;
    } else if (len > max[2]) {
      max[2] = len;
    }
  }
  if (max[0].first + max[1].first <= limit) {
    smin(h[x][0], max[0].first);
  }
  for (auto &v : max) {
    LL tmp = v.first;
    int i = v.second;
    int y = g[x][i].first, z = g[x][i].second;
    v.first = std::min(h[y][0] + std::max(0, z - a[x] - a[y]), h[y][1] + std::max(0, z - a[x]));
    if (std::max({max[0].first + max[1].first, max[0].first + max[2].first, max[1].first + max[2].first}) <= limit) {
      smin(h[x][1], std::max({max[0].first, max[1].first, max[2].first}));
    }
    v.first = tmp;
  }
  return std::min(h[x][0], h[x][1]) <= limit;
}
void solve() {
  std::cin >> n;
  for (int i = 1; i <= n; i++) {
    std::cin >> a[i];
    g[i].clear();

  for (int i = 1, x, y, z; i < n; i++) {
    std::cin >> x >> y >> z;
    g[x].push_back({y, z});
    g[y].push_back({x, z});

  while (l < r) {
    auto m = l + r >> 1;
    limit = m;
    if (dfs2(1, 0)) {
      r = m;
    } else {
      l = m + 1; 
    }
  }

}
