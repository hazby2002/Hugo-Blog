设每个边有没有为 0/1，列方程组解异或方程，求自由元个数。

// Author:  HolyK

#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;
constexpr int P(998244353);
inline void inc(int &x, int y) {
  x += y;
  if (x >= P) x -= P;
}
int fpow(int x, int k = P - 2) {
  int r = 1;
  for (; k; k >>= 1, x = 1LL * x * x % P) {
    if (k & 1) r = 1LL * r * x % P;
  }
  return r;
}
constexpr int N(17 * 16 * 2 + 5);
using BTS = std::bitset<N>;
BTS p[N], tmp;
bool ins(BTS &x) {
  for (int i = N - 1; i >= 0; i--) {
    if (!x[i]) continue;
    if (!p[i][i]) {
      p[i] = x;
      return true;
    }
    x ^= p[i];
  }
  return false;
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  std::cin >> t;
  while (t--) {
    int n, m;
    for (auto &v : p) v.reset();
    std::cin >> n >> m;
    n--, m--;
    int t = (n + 1) * m, ans = t + n * (m + 1);
    for (int i = 0; i < n; i++) {
      std::string s;
      std::cin >> s;
      for (int j = 0; j < m; j++) {
        if (s[j] == '.') continue;
        tmp.reset();
        tmp[i * m + j] = 1;
        tmp[(i + 1) * m + j] = 1;
        tmp[t + i * (m + 1) + j] = 1;
        tmp[t + i * (m + 1) + j + 1] = 1;
        tmp[N - 1] = s[j] - '0';
        if (ins(tmp)) ans--;
      }
    }
    for (int i = 0; i < n + 1; i++) {
      for (int j = 0; j < m + 1; j++) {
        tmp.reset();
        if (j < m) tmp[i * m + j] = 1;
        if (j) tmp[i * m + j - 1] = 1;
        if (i < n) tmp[t + i * (m + 1) + j] = 1;
        if (i) tmp[t + (i - 1) * (m + 1) + j] = 1;
        if (ins(tmp)) ans--;
      }
    }
    ans = fpow(2, ans);
    if (p[N - 1][N - 1]) {
      p[N - 1][N - 1] = 0;
      for (int i = N - 2; i >= 0; i--) {
        if (p[N - 1][i]) p[N - 1] ^= p[i];
      }
      if (p[N - 1].none()) ans = 0;
    }
    std::cout << ans << "\n";
  }
  return 0;
}

// Author:  HolyK
// Created: Tue Jul 20 12:34:29 2021
#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;

constexpr int N(1e5 + 5);

int ch[N * 30][2], max[N * 30], cnt;
void ins(int x, int id) {
  int o = 0;
  for (int i = 29; i >= 0; i--) {
    int u = x >> i & 1;
    if (!ch[o][u]) {
      ch[o][u] = ++cnt;
      ch[cnt][0] = ch[cnt][1] = 0;
      max[cnt] = 0;
    }
    o = ch[o][u];
    smax(max[o], id);
  }
  
}
PII ask(int x, int o = 0, int d = 29) {
  int ans = 0;
  for (int i = d; i >= 0; i--) {
    int u = x >> i & 1;
    if (ch[o][u ^ 1]) {
      ans |= 1 << i;
      o = ch[o][u ^ 1];
    } else {
      o = ch[o][u];
    }
  }
  return {ans, max[o]};
}
int getMax(int x, int k) {
  int ans = 0, o = 0, now = 0;
  for (int i = 29; i >= 0; i--) {
    auto v = ask(x, o, i);
    if (now + v.first >= k) {
      smax(ans, v.second);
    }
    int u = (x ^ k) >> i & 1;
    if (!ch[o][u]) {
      return ans;
      // u ^= 1;
    }
    if (k >> i & 1) now |= 1 << i;
    o = ch[o][u];
  }
  if (now >= k) smax(ans, max[o]);
  return ans;
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  std::cin >> t;
  while (t--) {
    int n, k;
    cnt = 0;
    ch[0][0] = ch[0][1] = 0;
    max[0] = 0;
    std::cin >> n >> k;
    ins(0, 0);
    int ans = 1e9, l = 0, r = 0;
    for (int i = 1, w = 0, x; i <= n; i++) {
      std::cin >> x;
      w ^= x;
      if (ask(w).first >= k) {
        x = getMax(w, k);
        if (smin(ans, i - x)) {
          l = x + 1;
          r = i;
        }
      }
      ins(w, i);
    }
    if (l) {
      std::cout << l << " " << r << "\n";
    } else {
      std::cout << "-1\n";
    }
    
  };
  return 0;
}

#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;
constexpr int P(998244353);
inline void inc(int &x, int y) {
  x += y;
  if (x >= P) x -= P;
}
int fpow(int x, int k = P - 2) {
  int r = 1;
  for (; k; k >>= 1, x = 1LL * x * x % P) {
    if (k & 1) r = 1LL * r * x % P;
  }
  return r;
}
int cal(int n, int t) {
  int ans = fpow(n - 1, t);
  if (t & 1) inc(ans, P - n + 1);
  else inc(ans, n - 1);
  return 1LL * ans * fpow(n) % P;
}

int bsgs(int a, int b) {
  b %= P;
  std::vector<PII> v;
  const int t = (int)sqrt(P) + 1;
  for (int j = 0, pw = 1; j < t; j++, pw = 1LL * pw * a % P) {
    int val = 1LL * b * pw % P;
    v.push_back({val, j});
  }
  std::sort(v.begin(), v.end());
  a = fpow(a, t);
  if (!a) return b == 0 ? 1 : -1;
  for (int i = 0, j, pw = 1; i <= t; i++, pw = 1LL * pw * a % P) {
    auto it = std::lower_bound(v.begin(), v.end(), PII(pw, 0));
    if (it == v.end() || it->first != pw) {
      j = -1;
    } else {
      j = it->second;
    }
    if (j >= 0 && i * t - j >= 0) return i * t - j;
  }
  return -1;
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  // std::cout << cal(11, 2);
  int t;
  std::cin >> t;
  while (t--) {
    int n, x, y;
    std::cin >> n >> x;
    y = x = 1LL * n * x % P;
    inc(x, n - 1);
    inc(y, P - n + 1);
    int t1 = bsgs(n - 1, x);
    int t2 = bsgs(n - 1, y);
    int ans = P;
    if (~t1 && (t1 & 1)) smin(ans, t1);
    if (~t2 && (~t2 & 1)) smin(ans, t2);
    if (ans == P) ans = -1;
    std::cout << ans << "\n";
  }
  return 0;
}

#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}
inline char gc() {
  static constexpr int BufferSize = 1 << 22 | 5;
  static char buf[BufferSize], *p, *q;
  static std::streambuf *i = std::cin.rdbuf();
  return p == q ? p = buf, q = p + i->sgetn(p, BufferSize), p == q ? EOF : *p++ : *p++;
}
struct Reader {
  template <class T>
  Reader &operator>>(T &w) {
    char c, p = 0;
    for (; !std::isdigit(c = gc());) if (c == '-') p = 1;
    for (w = c & 15; std::isdigit(c = gc()); w = w * 10 + (c & 15)) ;
    if (p) w = -w;
    return *this;
  }
} cin;

using LL = long long;
using PII = std::pair<int, int>;
constexpr int N(1e5 + 5), M(5e5 + 5);
struct Edge {
  int v, w, next;
} e[M << 1];
int head[N], tot;
void add(int x, int y, int z) {
  e[++tot] = {y, z, head[x]}, head[x] = tot;
  e[++tot] = {x, z, head[y]}, head[y] = tot;
}
int vis[N], cnt;
void dfs(int x, int d) {
  vis[x] = cnt;
  for (int i = head[x]; i; i = e[i].next) {
    int y = e[i].v;
    if (e[i].w > d || vis[y] == cnt) continue;
    dfs(y, d);
  }
}
std::array<int, 3> ee[M];
struct Dsu {
  std::vector<int> f;
  Dsu(int n) : f(n) {
    std::iota(f.begin(), f.end(), 0);
  }
  int find(int x) {
    while (x != f[x]) {
      x = f[x] = f[f[x]];
    }
    return x;
  }
  bool same(int x, int y) {
    return find(x) == find(y);
  }
  bool merge(int x, int y) {
    return same(x, y) ? false : f[find(x)] = find(y), true;
  }
};
int main() {
  // freopen("1.in", "r", stdin);
  // freopen("t.out", "w", stdout);
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  cin >> t;
  while (t--) {
    int n, m, k;
    cin >> n >> m >> k;
    int r = 0;
    for (int i = 1; i <= n; i++) head[i] = 0;
    tot = 0;
    for (int i = 1; i <= m; i++) {
      int x, y, z;
      cin >> x >> y >> z;
      ee[i] = {z, x, y};
      smax(r, z);
    }
    std::sort(ee + 1, ee + 1 + m);
    Dsu d(n + 1);
    for (int i = 1; i <= m; i++) {
      int x = ee[i][1], y = ee[i][2], z = ee[i][0];
      if (!d.same(x, y)) d.merge(x, y), add(x, y, z);
    }
    int l = 0;
    auto cal = [&](int d) {
      cnt++;
      int res = 0;
      for (int i = 1; i <= n; i++) {
        if (vis[i] != cnt) dfs(i, d), res++;
      }
      return res;
    };
    while (l < r) {
      int m = l + r >> 1;
      if (cal(m) > k) {
        l = m + 1;
      } else {
        r = m;
      }
    }
    if (cal(l) == k) {
      std::cout << l << "\n";
    } else {
      std::cout << "-1\n";
    }
  }
  return 0;
}

// Author:  HolyK
// Created: Tue Jul 20 12:11:35 2021
#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;

constexpr int N(1e5 + 5), T(512);
int n, m, a[N];
struct Qry {
  int l, r, x, y, id;
  bool operator<(const Qry &rhs) const {
    return l / T == rhs.l / T ? r < rhs.r : l < rhs.l;
  }
} q[N];
int sum[N], b[N];
void add(int p, int x) {
  b[p] += x;
  sum[p / T] += x;
}
int ask(int l, int r) {
  int bl = l / T, br = r / T, ans = 0;
  if (bl == br) {
    for (int i = l; i <= r; i++) ans += b[i];
    return ans;
  }
  int y = bl * T + T, x = br * T;
  for (int i = l; i < y; i++) ans += b[i];
  for (int i = x; i <= r; i++) ans += b[i];
  for (int i = bl + 1; i < br; i++) ans += sum[i];
  return ans;
}
int cnt[N], ans[N];
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  std::cin >> t;
  while (t--) {
    std::cin >> n >> m;
    for (int i = 1; i <= n; i++) std::cin >> a[i];
    for (int i = 1; i <= m; i++) {
      int a, b, c, d;
      std::cin >> a >> b >> c >> d;
      q[i] = {a, c, b, d, i};
    }
    std::sort(q + 1, q + 1 + m);
    int x = 1, y = 0;
    memset(cnt, 0, sizeof cnt);
    memset(sum, 0, sizeof sum);
    memset(b, 0, sizeof b);
    for (int i = 1; i <= m; i++) {
      int l, r, qx, qy, id;
      l = q[i].l;
      r = q[i].r;
      qx = q[i].x;
      qy = q[i].y;
      id = q[i].id;
      while (x > l) {
        if (++cnt[a[--x]] == 1) {
          add(a[x], 1);
        }
      }
      while (y < r) {
        if (++cnt[a[++y]] == 1) {
          add(a[y], 1);
        }
      }
      while (x < l) {
        if (--cnt[a[x]] == 0) {
          add(a[x], -1);
        }
        x++;
      }
      while (y > r) {
        if (--cnt[a[y]] == 0) {
          add(a[y], -1);
        }
        y--;
      }
      ans[id] = ask(qx, qy);
    }
    for (int i = 1; i <= m; i++) {
      std::cout << ans[i] << "\n";
    }
  }
  return 0;
}
