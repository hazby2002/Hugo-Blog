Display Substring

给定一个字符串 S，每个字符有固定的权值，子串的权值是字符权值和。求权值第 \(k\) 大的本质不同子串。

\(|S| \leq 10^5\)。

二分答案，然后后缀树每个节点二分一下长度。

// Author:  HolyK
// Created: Thu Jul 29 18:51:26 2021
#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;
constexpr int N(2e5 + 5);
std::array<int, 26> ch[N], val;
int len[N], fa[N] = {-1}, cnt, pos[N]; // fa[0] = -1
int ins(int last, int x) {
  if (ch[last][x] && len[last] + 1 == len[ch[last][x]]) return ch[last][x];
  int p = last, np = ++cnt, q;
  ch[np].fill(0);
  len[np] = len[p] + 1;
  for (; ~p && !ch[p][x]; p = fa[p]) ch[p][x] = np;
  if (p == -1) {
    fa[np] = 0;
  } else if (len[q = ch[p][x]] == len[p] + 1) {
    fa[np] = q;
  } else {
    int nq = p == last ? np : ++cnt;
    ch[nq].fill(0);
    len[nq] = len[p] + 1;
    ch[nq] = ch[q];
    pos[nq] = pos[q];
    for (; ~p && ch[p][x] == q; p = fa[p]) ch[p][x] = nq;
    fa[np] = nq;
    fa[nq] = fa[q];
    fa[q] = nq;    
  }
  return np;
}
int sum[N];
int cal(int p, int x) {
  int l = len[fa[p]], r = len[p], i = pos[p];
  while (l < r) {
    int m = l + r + 1 >> 1;
    if (sum[i] - sum[i - m] <= x) {
      l = m;
    } else {
      r = m - 1;
    }
  }
  return l - len[fa[p]];
}

void solve() {
  int n;
  LL k;
  std::string s;
  std::cin >> n >> k >> s;
  for (int i = 0; i < 26; i++) std::cin >> val[i];
  cnt = 0;
  ch[0].fill(0);
  for (int last = 0, i = 0; i < n; i++) {
    last = ins(last, s[i] - 'a');
    pos[last] = i + 1;
  }
  for (int i = 0; i < n; i++) sum[i + 1] = sum[i] + val[s[i] - 'a'];
  LL c = 0;
  for (int i = 1; i <= cnt; i++) c += len[i] - len[fa[i]];
  if (c < k) {
    std::cout << "-1\n";
    return;
  }
  int l = 0, r = 1e7;
  while (l < r) {
    int m = l + r >> 1;
    c = 0;
    for (int i = 1; i <= cnt; i++) {
      c += cal(i, m);
      if (c >= k) break;
    }
    if (c < k) {
      l = m + 1;
    } else {
      r = m;
    }
  }
  std::cout << l << "\n";
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  std::cin >> t;
  while (t--) {
    solve();
  }
  return 0;
}

Didn’t I Say to Make My Abilities Average in the Next Life?!

多次询问区间的子区间最大最小值的平均值的期望。 \(10^5\)

原题 [HNOI2016]序列。

单调栈维护最大最小值，线段树维护历史和。

// Author:  HolyK
// Created: Thu Jul 29 15:51:30 2021
#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;
constexpr int P(1e9 + 7);
inline void inc(int &x, int y) {
  x += y;
  if (x >= P) x -= P;
}
int fpow(int x, int k = P - 2) {
  int r = 1;
  for (; k; k >>= 1, x = 1LL * x * x % P) {
    if (k & 1) r = 1LL * r * x % P;
  }
  return r;
}
constexpr int N(2e5 + 5);
#define ls o << 1
#define rs o << 1 | 1
int s1[N << 2], s2[N << 2], t1[N << 2], t2[N << 2];
void update(int o, int l, int r, int x, int y, int z, int i) {
  s1[o] = (s1[o] + (y - x + 1LL) * z) % P;
  s2[o] = (s2[o] + (y - x + 1LL) * z % P * i) % P;
  if (x <= l && r <= y) {
    inc(t1[o], z);
    t2[o] = (t2[o] + 1LL * z * i) % P;
    return;
  }
  int m = l + r >> 1;
  if (y <= m) return update(ls, l, m, x, y, z, i);
  if (x > m) return update(rs, m + 1, r, x, y, z, i);
  update(ls, l, m, x, m, z, i);
  update(rs, m + 1, r, m + 1, y, z, i);
}
int ask(int o, int l, int r, int x, int y, int i) {
  if (x <= l && r <= y) {
    return (1LL * i * s1[o] - s2[o] + P) % P;
  }
  int m = l + r >> 1, ans = (y - x + 1LL) * (1LL * t1[o] * i % P - t2[o] + P) % P;
  if (y <= m) {
    inc(ans, ask(ls, l, m, x, y, i));
  } else if (x > m) {
    inc(ans, ask(rs, m + 1, r, x, y, i));
  } else {
    inc(ans, ask(ls, l, m, x, m, i));
    inc(ans, ask(rs, m + 1, r, m + 1, y, i));
  }
  return ans;
}
int n, m, a[N], max[N], min[N], tmax, tmin;
std::vector<PII> g[N];
LL ans[N];
void solve() {
  std::cin >> n >> m;
  for (int i = 1; i <= n; i++) {
    std::cin >> a[i];
    g[i].clear();
  }
  for (int i = 1; i <= m; i++) {
    int x, y;
    std::cin >> x >> y;
    g[y].push_back({x, i});   
  }
  for (int i = 1; i <= n << 2; i++) {
    s1[i] = s2[i] = t1[i] = t2[i] = 0;
  }
  tmax = tmin = 0;
  for (int i = 1; i <= n; i++) {
    for (; tmax && a[max[tmax]] < a[i]; tmax--) {
      update(1, 1, n, max[tmax - 1] + 1, max[tmax], P - a[max[tmax]], i);
    }
    for (; tmin && a[min[tmin]] > a[i]; tmin--) {
      update(1, 1, n, min[tmin - 1] + 1, min[tmin], P - a[min[tmin]], i);
    }
    update(1, 1, n, max[tmax] + 1, i, a[i], i);
    update(1, 1, n, min[tmin] + 1, i, a[i], i);
    max[++tmax] = min[++tmin] = i;
    for (auto &v : g[i]) {
      int x = v.first, y = i, id = v.second;
      ans[id] = (P + 1LL >> 1) * ask(1, 1, n, x, y, i + 1) % P;
      ans[id] = 1LL * fpow(1LL * (y - x + 1) * (y - x + 2) / 2 % P) * ans[id] % P;
    }
  }
  for (int i = 1; i <= m; i++) {
    std::cout << (ans[i] + P) % P << "\n";
  }
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  std::cin >> t;
  while (t--) {
    solve();
  }
  return 0;
}

Directed Minimum Spanning Tree

Increasing Subsequence

给定 \(1\dots n\) 的排列 \(a_n\)，求极长上升子序列个数。一个上升子序列是极长上升子序列，当且仅当它不被其他上升子序列包含。

\(n \leq 10^5\)。

SegmentTree Beats

赛时做法。

设 \(dp_i\) 表示考虑了前 \(i\) 个位置，以 \(a_i\) 结尾的极长上升子序列个数。

发现 \(j\) 可以转移到 \(i\) 需要满足下面三个条件： \[ \begin{cases} j < i,\\ a_j < a_i,\\ \nexists k\in(j, i), a_j < a_k < a_i \end{cases} \]

考虑将 \(a_i\) 从小到大插入序列，去掉了第二个条件。

对于第三个条件，设 \(next_j\) 表示右边第一个比 \(a_j\) 大的位置，那么第三个条件转化为 \(next_j > i\)，插入 \(i\) 后修改 \(next_j = i\)。

可以发现这是区间取 \(\min\) 操作，在线段树上维护最大值和次大值以及最大值的 \(dp\) 值之和即可。

只有区间取 \(\min\)，因此复杂度是 \(O(n \log n)\)。

// Author:  HolyK
// Created: Thu Jul 29 16:41:11 2021
#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;
constexpr int P(998244353);
inline void inc(int &x, int y) {
  x += y;
  if (x >= P) x -= P;
}
int fpow(int x, int k = P - 2) {
  int r = 1;
  for (; k; k >>= 1, x = 1LL * x * x % P) {
    if (k & 1) r = 1LL * r * x % P;
  }
  return r;
}
constexpr int N(2e5 + 5);
#define ls o << 1
#define rs o << 1 | 1
int sumMod(int x, int y) {
  return x + y >= P ? x + y - P : x + y;
}
int max1[N << 2], max2[N << 2], cnt[N << 2], tag[N << 2];
void pushup(int o) {
  if (max1[ls] == max1[rs]) {
    max1[o] = max1[rs];
    max2[o] = std::max(max2[ls], max2[rs]);
    cnt[o] = cnt[ls];
    inc(cnt[o], cnt[rs]);
  } else if (max1[ls] > max1[rs]) {
    max1[o] = max1[ls];
    max2[o] = std::max(max2[ls], max1[rs]);
    cnt[o] = cnt[ls];
  } else {
    max1[o] = max1[rs];
    max2[o] = std::max(max1[ls], max2[rs]);
    cnt[o] = cnt[rs];
  }
}
void pushtag(int o, int tg) {
  if (max1[o] <= tg) return;
  max1[o] = tag[o] = tg;
}
void pushdown(int o) {
  if (tag[o] == -1) return;
  pushtag(ls, tag[o]);
  pushtag(rs, tag[o]);
  tag[o] = -1;
}
int update(int o, int l, int r, int x, int y, int z) {
  if (max1[o] <= z) return 0;
  if (x <= l && r <= y && max2[o] <= z) {
    int res = cnt[o];
    pushtag(o, z);
    return res;
  }
  int m = l + r >> 1;
  pushdown(o);
  int res;
  if (y <= m) res = update(ls, l, m, x, y, z);
  else if (x > m) res = update(rs, m + 1, r, x, y, z);
  else res = sumMod(update(ls, l, m, x, y, z), update(rs, m + 1, r, x, y, z));
  pushup(o);
  return res;
}
int val[N];
void update(int o, int l, int r, int x) {
  if (l == r) {
    max1[o] = 1e9;
    max2[o] = 0;
    cnt[o] = val[l];
    return;
  }
  int m = l + r >> 1;
  pushdown(o);
  x <= m ? update(ls, l, m, x) : update(rs, m + 1, r, x);
  pushup(o);
}
void build(int o, int l, int r) {
  tag[o] = -1;
  if (l == r) {
    max1[o] = max2[o] = 0;
    cnt[o] = 0;
    return;
  }
  int m = l + r >> 1;
  build(ls, l, m);
  build(rs, m + 1, r);
  pushup(o);
}
int a[N], p[N];
void solve() {
  int n;
  std::cin >> n;
  build(1, 1, n);
  for (int i = 1; i <= n; i++) std::cin >> a[i], p[a[i]] = i, val[i] = 0;
  for (int i = 1; i <= n; i++) {
    int j = p[i];
    if (j > 1) val[j] = update(1, 1, n, 1, j - 1, j);
    smax(val[j], 1);
    update(1, 1, n, j);
  }
  int ans = 0;
  for (int i = n, m = 0; i; i--) {
    if (smax(m, a[i])) {
      inc(ans, val[i]);
    }
  }
  std::cout << ans << "\n";
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  std::cin >> t;
  while (t--) {
    solve();
  }
  return 0;
}

cdq分治

题解做法。

分治，考虑 \([l, m)\) 对 \([m, r)\) 的贡献。

对于 \(i \in [m, r), j \in [l, m)\)，考虑按 \(a\) 从小到大插入，\(j\) 对 \(i\) 有贡献的条件是右边不能有比 \(a_j\) 大的数。

对于区间 \([l, m)\)，插入一个数 \(a_j\) 后，下标小于 \(j\) 的已插入的数都不能产生贡献，因此要把它们删除，用下标递减的单调栈维护这个操作。

对于区间 \([m, i)\)，设 \(k \in [m, i), a_k < a_i\)，则 \(j\) 要满足 \(a_j > \max a_k\)，\(\max a_k\) 同样可以用单调栈维护。

注意到 \([l, m)\) 的单调栈同时也是值递增的，所以可以二分找到对应位置，然后用前缀和求出答案。

复杂度 \(O(n \log^2 n)\)，这里写的是非递归的，实际表现不劣于线段树做法。

// Author:  HolyK
// Created: Fri Jul 30 11:32:47 2021
#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;
constexpr int P(998244353);
inline void inc(int &x, int y) {
  x += y;
  if (x >= P) x -= P;
}
int fpow(int x, int k = P - 2) {
  int r = 1;
  for (; k; k >>= 1, x = 1LL * x * x % P) {
    if (k & 1) r = 1LL * r * x % P;
  }
  return r;
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  std::cin >> t;
  while (t--) {
    int n;
    std::cin >> n;
    std::vector<int> a(n), b, f(n), s1, s2, sum;
    for (int i = 0, m = n + 1; i < n; i++) {
      std::cin >> a[i];
      if (smin(m, a[i])) f[i] = 1;
    }
    for (int m = 1; m < n; m++) {
      int l = m & m - 1, r = std::min(n, m + (m & -m));
      b.resize(r - l);
      std::iota(b.begin(), b.end(), l);
      std::sort(b.begin(), b.end(), [&](int i, int j) {
        return a[i] < a[j];
      });
      s1.clear(), s2.clear(), sum.clear();
      for (int j = 0; j < r - l; j++) {
        int i = b[j];
        if (i < m) {
          while (!s1.empty() && b[s1.back()] < i) s1.pop_back(), sum.pop_back();
          s1.push_back(j);
          sum.push_back(f[i]);
          if (sum.size() > 1) inc(sum.back(), sum[sum.size() - 2]);
        } else {
          while (!s2.empty() && b[s2.back()] > i) s2.pop_back();
          if (!s1.empty()) {
            int k = std::lower_bound(s1.begin(), s1.end(), s2.empty() ? -1 : s2.back()) - s1.begin();
            inc(f[i], sum.back());
            if (k) inc(f[i], P - sum[k - 1]);
          }
          s2.push_back(j);
        }
      }
    }
    int ans = 0;
    for (int i = n - 1, m = 0; i >= 0; i--) {
      if (smax(m, a[i])) inc(ans, f[i]);
    }
    std::cout << ans << "\n";
  }
  return 0;
}

Lawn of the Dead

// Author:  HolyK
// Created: Thu Jul 29 14:00:33 2021
#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;
inline char gc() {
  static constexpr int BufferSize = 1 << 22 | 5;
  static char buf[BufferSize], *p, *q;
  static std::streambuf *i = std::cin.rdbuf();
  return p == q ? p = buf, q = p + i->sgetn(p, BufferSize), p == q ? EOF : *p++ : *p++;
}
struct Reader {
  template <class T>
  Reader &operator>>(T &w) {
    char c, p = 0;
    for (; !std::isdigit(c = gc());) if (c == '-') p = 1;
    for (w = c & 15; std::isdigit(c = gc()); w = w * 10 + (c & 15)) ;
    if (p) w = -w;
    return *this;
  }
} cin;

constexpr int N(1e5 + 5);
struct Info {
  int pre, sum, len;
};
Info merge(Info a, Info b) {
  return {
    a.pre == a.len ? a.len + b.pre : a.pre,
    a.sum + b.sum,
    a.len + b.len
  };
}
#define ls o << 1
#define rs o << 1 | 1
Info t[N << 2];
int tag[N << 2];

void update(int o, int l, int r, int z) {
  tag[o] = z;
  if (z == 0) {
    t[o] = {r - l + 1, 0, r - l + 1};
  } else {
    t[o] = {0, r - l + 1, r - l + 1};
  }
}
void pushdown(int o, int l, int r) {
  if (tag[o] != -1) {
    int m = l + r >> 1;
    update(ls, l, m, tag[o]);
    update(rs, m + 1, r, tag[o]);
    tag[o] = -1;
  }
}
void update(int o, int l, int r, int x, int y, int z) {
  if (x > y || r < x || y < l) return;
  if (x <= l && r <= y) {
    update(o, l, r, z);
    return;
  }
  int m = l + r >> 1;
  pushdown(o, l, r);
  if (x <= m) update(ls, l, m, x, y, z);
  if (y > m) update(rs, m + 1, r, x, y, z);
  t[o] = merge(t[ls], t[rs]);
}
Info ask(int o, int l, int r, int x, int y) {

}

    return;
  }
  int m = l + r >> 1;
  build(ls, l, m), build(rs, m + 1, r);
  t[o] = merge(t[ls], t[rs]);
}
