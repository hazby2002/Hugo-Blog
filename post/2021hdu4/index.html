Display Substring

// Author:  HolyK
// Created: Thu Jul 29 18:51:26 2021
#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;
constexpr int N(2e5 + 5);
std::array<int, 26> ch[N], val;
int len[N], fa[N] = {-1}, cnt, pos[N]; // fa[0] = -1
int ins(int last, int x) {
  if (ch[last][x] && len[last] + 1 == len[ch[last][x]]) return ch[last][x];
  int p = last, np = ++cnt, q;
  ch[np].fill(0);
  len[np] = len[p] + 1;
  for (; ~p && !ch[p][x]; p = fa[p]) ch[p][x] = np;
  if (p == -1) {
    fa[np] = 0;
  } else if (len[q = ch[p][x]] == len[p] + 1) {
    fa[np] = q;
  } else {
    int nq = p == last ? np : ++cnt;
    ch[nq].fill(0);
    len[nq] = len[p] + 1;
    ch[nq] = ch[q];
    pos[nq] = pos[q];
    for (; ~p && ch[p][x] == q; p = fa[p]) ch[p][x] = nq;
    fa[np] = nq;
    fa[nq] = fa[q];
    fa[q] = nq;    
  }
  return np;
}
int sum[N];
int cal(int p, int x) {
  int l = len[fa[p]], r = len[p], i = pos[p];
  while (l < r) {
    int m = l + r + 1 >> 1;
    if (sum[i] - sum[i - m] <= x) {
      l = m;
    } else {
      r = m - 1;
    }
  }
  return l - len[fa[p]];
}

void solve() {
  int n;
  LL k;
  std::string s;
  std::cin >> n >> k >> s;
  for (int i = 0; i < 26; i++) std::cin >> val[i];
  cnt = 0;
  ch[0].fill(0);
  for (int last = 0, i = 0; i < n; i++) {
    last = ins(last, s[i] - 'a');
    pos[last] = i + 1;
  }
  for (int i = 0; i < n; i++) sum[i + 1] = sum[i] + val[s[i] - 'a'];
  LL c = 0;
  for (int i = 1; i <= cnt; i++) c += len[i] - len[fa[i]];
  if (c < k) {
    std::cout << "-1\n";
    return;
  }
  int l = 0, r = 1e7;
  while (l < r) {
    int m = l + r >> 1;
    c = 0;
    for (int i = 1; i <= cnt; i++) {
      c += cal(i, m);
      if (c >= k) break;
    }
    if (c < k) {
      l = m + 1;
    } else {
      r = m;
    }
  }
  std::cout << l << "\n";
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  std::cin >> t;
  while (t--) {
    solve();
  }
  return 0;
}

Didn’t I Say to Make My Abilities Average in the Next Life?!

单调栈维护最大最小值，线段树维护历史和。

// Author:  HolyK
// Created: Thu Jul 29 15:51:30 2021
#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;
constexpr int P(1e9 + 7);
inline void inc(int &x, int y) {
  x += y;
  if (x >= P) x -= P;
}
int fpow(int x, int k = P - 2) {
  int r = 1;
  for (; k; k >>= 1, x = 1LL * x * x % P) {
    if (k & 1) r = 1LL * r * x % P;
  }
  return r;
}
constexpr int N(2e5 + 5);
#define ls o << 1
#define rs o << 1 | 1
int s1[N << 2], s2[N << 2], t1[N << 2], t2[N << 2];
void update(int o, int l, int r, int x, int y, int z, int i) {
  s1[o] = (s1[o] + (y - x + 1LL) * z) % P;
  s2[o] = (s2[o] + (y - x + 1LL) * z % P * i) % P;
  if (x <= l && r <= y) {
    inc(t1[o], z);
    t2[o] = (t2[o] + 1LL * z * i) % P;
    return;
  }
  int m = l + r >> 1;
  if (y <= m) return update(ls, l, m, x, y, z, i);
  if (x > m) return update(rs, m + 1, r, x, y, z, i);
  update(ls, l, m, x, m, z, i);
  update(rs, m + 1, r, m + 1, y, z, i);
}
int ask(int o, int l, int r, int x, int y, int i) {
  if (x <= l && r <= y) {
    return (1LL * i * s1[o] - s2[o] + P) % P;
  }
  int m = l + r >> 1, ans = (y - x + 1LL) * (1LL * t1[o] * i % P - t2[o] + P) % P;
  if (y <= m) {
    inc(ans, ask(ls, l, m, x, y, i));
  } else if (x > m) {
    inc(ans, ask(rs, m + 1, r, x, y, i));
  } else {
    inc(ans, ask(ls, l, m, x, m, i));
    inc(ans, ask(rs, m + 1, r, m + 1, y, i));
  }
  return ans;
}
int n, m, a[N], max[N], min[N], tmax, tmin;
std::vector<PII> g[N];
LL ans[N];
void solve() {
  std::cin >> n >> m;
  for (int i = 1; i <= n; i++) {
    std::cin >> a[i];
    g[i].clear();
  }
  for (int i = 1; i <= m; i++) {
    int x, y;
    std::cin >> x >> y;
    g[y].push_back({x, i});   
  }
  for (int i = 1; i <= n << 2; i++) {
    s1[i] = s2[i] = t1[i] = t2[i] = 0;
  }
  tmax = tmin = 0;
  for (int i = 1; i <= n; i++) {
    for (; tmax && a[max[tmax]] < a[i]; tmax--) {
      update(1, 1, n, max[tmax - 1] + 1, max[tmax], P - a[max[tmax]], i);
    }
    for (; tmin && a[min[tmin]] > a[i]; tmin--) {
      update(1, 1, n, min[tmin - 1] + 1, min[tmin], P - a[min[tmin]], i);
    }
    update(1, 1, n, max[tmax] + 1, i, a[i], i);
    update(1, 1, n, min[tmin] + 1, i, a[i], i);
    max[++tmax] = min[++tmin] = i;
    for (auto &v : g[i]) {
      int x = v.first, y = i, id = v.second;
      ans[id] = (P + 1LL >> 1) * ask(1, 1, n, x, y, i + 1) % P;
      ans[id] = 1LL * fpow(1LL * (y - x + 1) * (y - x + 2) / 2 % P) * ans[id] % P;
    }
  }
  for (int i = 1; i <= m; i++) {
    std::cout << (ans[i] + P) % P << "\n";
  }
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  std::cin >> t;
  while (t--) {
    solve();
  }
  return 0;
}

Increasing Subsequence

SegmentTree Beats!

// Author:  HolyK
// Created: Thu Jul 29 16:41:11 2021
#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;
constexpr int P(998244353);
inline void inc(int &x, int y) {
  x += y;
  if (x >= P) x -= P;
}
int fpow(int x, int k = P - 2) {
  int r = 1;
  for (; k; k >>= 1, x = 1LL * x * x % P) {
    if (k & 1) r = 1LL * r * x % P;
  }
  return r;
}
constexpr int N(2e5 + 5);
#define ls o << 1
#define rs o << 1 | 1
int sumMod(int x, int y) {
  return x + y >= P ? x + y - P : x + y;
}
int max1[N << 2], max2[N << 2], cnt[N << 2], tag[N << 2];
void pushup(int o) {
  if (max1[ls] == max1[rs]) {
    max1[o] = max1[rs];
    max2[o] = std::max(max2[ls], max2[rs]);
    cnt[o] = cnt[ls];
    inc(cnt[o], cnt[rs]);
  } else if (max1[ls] > max1[rs]) {
    max1[o] = max1[ls];
    max2[o] = std::max(max2[ls], max1[rs]);
    cnt[o] = cnt[ls];
  } else {
    max1[o] = max1[rs];
    max2[o] = std::max(max1[ls], max2[rs]);
    cnt[o] = cnt[rs];
  }
}
void pushtag(int o, int tg) {
  if (max1[o] <= tg) return;
  max1[o] = tag[o] = tg;
}
void pushdown(int o) {
  if (tag[o] == -1) return;
  pushtag(ls, tag[o]);
  pushtag(rs, tag[o]);
  tag[o] = -1;
}
int update(int o, int l, int r, int x, int y, int z) {
  if (max1[o] <= z) return 0;
  if (x <= l && r <= y && max2[o] <= z) {
    int res = cnt[o];
    pushtag(o, z);
    return res;
  }
  int m = l + r >> 1;
  pushdown(o);
  int res;
  if (y <= m) res = update(ls, l, m, x, y, z);
  else if (x > m) res = update(rs, m + 1, r, x, y, z);
  else res = sumMod(update(ls, l, m, x, y, z), update(rs, m + 1, r, x, y, z));
  pushup(o);
  return res;
}
int val[N];
void update(int o, int l, int r, int x) {
  if (l == r) {
    max1[o] = 1e9;
    max2[o] = 0;
    cnt[o] = val[l];
    return;
  }
  int m = l + r >> 1;
  pushdown(o);
  x <= m ? update(ls, l, m, x) : update(rs, m + 1, r, x);
  pushup(o);
}
void build(int o, int l, int r) {
  tag[o] = -1;
  if (l == r) {
    max1[o] = max2[o] = 0;
    cnt[o] = 0;
    return;
  }
  int m = l + r >> 1;
  build(ls, l, m);
  build(rs, m + 1, r);
  pushup(o);
}
int a[N], p[N];
void solve() {
  int n;
  std::cin >> n;
  build(1, 1, n);
  for (int i = 1; i <= n; i++) std::cin >> a[i], p[a[i]] = i, val[i] = 0;
  for (int i = 1; i <= n; i++) {
    int j = p[i];
    if (j > 1) val[j] = update(1, 1, n, 1, j - 1, j);
    smax(val[j], 1);
    update(1, 1, n, j);
  }
  int ans = 0;
  for (int i = n, m = 0; i; i--) {
    if (smax(m, a[i])) {
      inc(ans, val[i]);
    }
  }
  std::cout << ans << "\n";
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  std::cin >> t;
  while (t--) {
    solve();
  }
  return 0;
}

// Author:  HolyK

#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;
inline char gc() {
  static constexpr int BufferSize = 1 << 22 | 5;
  static char buf[BufferSize], *p, *q;
  static std::streambuf *i = std::cin.rdbuf();
  return p == q ? p = buf, q = p + i->sgetn(p, BufferSize), p == q ? EOF : *p++ : *p++;
}
struct Reader {
  template <class T>
  Reader &operator>>(T &w) {
    char c, p = 0;
    for (; !std::isdigit(c = gc());) if (c == '-') p = 1;
    for (w = c & 15; std::isdigit(c = gc()); w = w * 10 + (c & 15)) ;
    if (p) w = -w;
    return *this;
  }
} cin;

constexpr int N(1e5 + 5);
struct Info {
  int pre, sum, len;
};
Info merge(Info a, Info b) {
  return {
    a.pre == a.len ? a.len + b.pre : a.pre,
    a.sum + b.sum,
    a.len + b.len
  };
}
#define ls o << 1
#define rs o << 1 | 1
Info t[N << 2];
int tag[N << 2];

void update(int o, int l, int r, int z) {
  tag[o] = z;
  if (z == 0) {
    t[o] = {r - l + 1, 0, r - l + 1};
  } else {
    t[o] = {0, r - l + 1, r - l + 1};
  }
}
void pushdown(int o, int l, int r) {
  if (tag[o] != -1) {
    int m = l + r >> 1;
    update(ls, l, m, tag[o]);
    update(rs, m + 1, r, tag[o]);
    tag[o] = -1;
  }
}
void update(int o, int l, int r, int x, int y, int z) {
  if (x > y || r < x || y < l) return;
  if (x <= l && r <= y) {
    update(o, l, r, z);
    return;
  }
  int m = l + r >> 1;
  pushdown(o, l, r);
  if (x <= m) update(ls, l, m, x, y, z);
  if (y > m) update(rs, m + 1, r, x, y, z);
  t[o] = merge(t[ls], t[rs]);
}
Info ask(int o, int l, int r, int x, int y) {
  
  if (x <= l && r <= y) return t[o];
  int m = l + r >> 1;
  pushdown(o, l, r);
  if (y <= m) return ask(ls, l, m, x, y);
  if (x > m) return ask(rs, m + 1, r, x, y);
  return merge(ask(ls, l, m, x, y), ask(rs, m + 1, r, x, y));
}
void build(int o, int l, int r) {
  tag[o] = -1;
  if (l == r) {
    if (l == 1) update(o, l, r, 1);
    else update(o, l, r, 0);
    return;
  }
  int m = l + r >> 1;
  build(ls, l, m), build(rs, m + 1, r);
  t[o] = merge(t[ls], t[rs]);
}
std::vector<int> g[N];
std::vector<PII> p;
int main() {
  // freopen("t.in", "r", stdin);
  // freopen("t.out", "w", stdout);
  double sta = (double)clock() / CLOCKS_PER_SEC;
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  cin >> t;
  while (t--) {
    int n, m, k;
    cin >> n >> m >> k;
    for (int i = 1; i <= n; i++) {
      g[i].clear();
    }
    while (k--) {
      int x, y;
      cin >> x >> y;
      g[x].push_back(y);
    }
    build(1, 1, m);
    LL ans = 0;
    for (int i = 1; i <= n; i++) {
      p.clear();
      for (auto y : g[i]) {
        // update(1, 1, m, y, y, 0);
        int r = y;
        if (y < m) r += ask(1, 1, m, y + 1, m).pre;
        p.push_back({y, r});
      }
      int l = ::t[1].pre;
      update(1, 1, m, 1, m, 1);
      update(1, 1, m, 1, l, 0);
      for (auto v : p) update(1, 1, m, std::max(l + 1, v.first), v.second, 0);
      ans += ::t[1].sum;
    }
    std::cout << ans << "\n";
  }
  // std::cerr << (double)clock() / CLOCKS_PER_SEC - sta << "\n";
  return 0;
}
