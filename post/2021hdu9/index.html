  int n;
  std::cin >> n;
  std::vector<int> a(n), b(n), p(n), q;
  for (int i = 0; i < n; i++) {
    std::cin >> a[i];
    p[i] = i;
  }
  std::sort(p.begin(), p.end(), [&](int i, int j) {
    return a[i] < a[j];
  });
  for (int i = 0; i < n; i++) {
    std::cin >> b[i];
  }
  std::sort(b.rbegin(), b.rend());
  q = p;
  std::reverse(q.begin(), q.end());
  for (int i = 0, j, x, ans; i < n; i++) {
    x = a[i] + b[0], ans = n, j = 1;
    for (int k : p) {
      if (k == i) continue;
      while (j < n && b[j] + a[k] > x) j++;
      if (j == n) break;
      j++;
      ans--;
    }
    std::cout << ans << " ";
    x = a[i] + b.back(), j = n - 2, ans = 1;
    for (int k : q) {
      if (k == i) continue;
      while (j >= 0 && b[j] + a[k] <= x) j--;
      if (j < 0) break;
      j--;
      ans++;
    }
    std::cout << ans << "\n";
  }
}

constexpr int N(2e5 + 5);
#define ls o << 1
#define rs o << 1 | 1
struct Node {
  int siz;
  LL s1, s2, t1, t2;
} t[N << 2];
void pushup(int o) {
  t[o] = {
    t[ls].siz + t[rs].siz,
    t[ls].s1 + t[rs].s1,
    t[ls].s2 + t[rs].s2,
    t[o].t1, t[o].t2
  };
}
void add1(int o, LL x) {
  t[o].s1 += t[o].siz * x;
  t[o].t1 += x;
}
void add2(int o, LL x) {
  t[o].s2 += t[o].siz * x;
  t[o].t2 += x;
}
void pushdown(int o) {
  if (t[o].t1) {
    add1(ls, t[o].t1), add1(rs, t[o].t1);
    t[o].t1 = 0;
  }
  if (t[o].t2) {
    add2(ls, t[o].t2), add2(rs, t[o].t2);
    t[o].t2 = 0;
  }
}
void update1(int o, int l, int r, int x, int y, int z) {
  if (x <= l && r <= y) {
    add1(o, z);
    return;
  }
  pushdown(o);
  int m = l + r >> 1;
  if (x <= m) update1(ls, l, m, x, y, z);
  if (y > m) update1(rs, m + 1, r, x, y, z);
  pushup(o);
}
void update2(int o, int l, int r, int x, int y, int z) {
  if (x <= l && r <= y) {
    add2(o, z);
    return;
  }
  pushdown(o);
  int m = l + r >> 1;
  if (x <= m) update2(ls, l, m, x, y, z);
  if (y > m) update2(rs, m + 1, r, x, y, z);
  pushup(o);
}

int a[N], tot;
void ins(int x, int y) {
  a[x] += y;
  tot += y;
  LL s1 = 0, s2 = 0;
  int o = 1, l = 1, r = 2e5;
  while (l < r) {
    pushdown(o);
    int m = l + r >> 1;
    if (x <= m) {
      r = m;
      s2 += t[rs].siz;
      o = ls;
    } else {
      l = m + 1;
      s1 += t[ls].siz;
      o = rs;
    }
  }
  t[o].siz += y;
  t[o].s1 += s1 * y;
  t[o].s2 += s2 * y;
  while (o >>= 1) pushup(o);
  if (x < 2e5) update1(1, 1, 2e5, x + 1, 2e5, y);
  if (1 <   x) update2(1, 1, 2e5, 1, x - 1, y);
}
LL ask() {
  int o = 1, l = 1, r = 2e5, s = 0;
  LL ans = 0;
  while (l < r) {
    pushdown(o);
    int m = l + r >> 1;
    if (2 * (s + t[ls].siz) - a[m] < tot) {
      s += t[ls].siz;
      ans += t[ls].s2;
      o = rs;
      l = m + 1;
    } else {
      ans += t[rs].s1;
      o = ls;
      r = m;
    }
  }
  // std::cerr << "?? " <<  l << " " << ans << " " << t[o].siz << " " << t[o].s1 << " " << t[o].s2 << "\n";
  if (s < tot - s - a[l]) {
    ans += t[o].s2;
  } else {
    ans += t[o].s1;
  }
  return ans;
  // return 0;
}

inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;
inline char gc() {
  static constexpr int BufferSize = 1 << 22 | 5;
  static char buf[BufferSize], *p, *q;
  static std::streambuf *i = std::cin.rdbuf();
  return p == q ? p = buf, q = p + i->sgetn(p, BufferSize), p == q ? EOF : *p++ : *p++;
}
struct Reader {
  template <class T>
  Reader &operator>>(T &w) {
    char c, p = 0;
    for (; !std::isdigit(c = gc());) if (c == '-') p = 1;

    return *this;
  }

constexpr int N(2e5 + 5);
int a[N], c[N], m;
LL d[N], m2;
void add(int p, int x) {
  LL y = 2LL * a[p] * x + 1LL * x * x;
  a[p] += x;
  m += x;
  m2 += y;
  for (; p <= 2e5; p += p & -p) {
    c[p] += x;
    d[p] += y;
  }
}
void solve() {
  memset(a, 0, sizeof a);
  memset(c, 0, sizeof c);
  memset(d, 0, sizeof d);
  m = m2 = 0;
  int n, q;
  cin >> n >> q;
  while (n--) {
    int x;
    cin >> x;
    add(x, 1);
  }
  while (q--) {
    int opt, x, y;
    cin >> opt;
    if (opt == 1) {
      cin >> x >> y;
      add(y, x);
    } else {
      int s = 0, s2 = 0, p = 0;
      LL ans = 0;
      for (int i = 1 << 17; i; i >>= 1) {
        if (p + i <= 2e5 && (s + c[p + i]) * 2 - a[p + i] < m) {
          p += i;
          s += c[p];
          s2 += d[p];
          ans += 1LL * c[p] * (m - s) + (1LL * c[p] * c[p] - d[p]) / 2;
        }
      }
      ans += 1LL * (m - s) * s + (1LL * (m - s) * (m - s) - m2 + s2) / 2;
      LL k = 1LL * m * (m - 1), g = std::__gcd(ans, k);
      std::cout << ans / g << "/" << k / g << "\n";
    }
  }
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int t;
  cin >> t;

    solve();
  }

int pre[N], next[N], siz[2], head[2], tail[2], pos[N];
void push_back(int p, int x) {
  pos[x] = p;
  int y = tail[p];
  pre[x] = y;
  if (y) next[y] = x;
  tail[p] = x;
  if (!head[p]) head[p] = x;
  siz[p]++;
}
void push_front(int p, int x) {
  pos[x] = p;
  int y = head[p];
  next[x] = y;
  if (y) pre[y] = x;
  head[p] = x;
  if (!tail[p]) tail[p] = x;
  siz[p]++;
}
int pop_front(int p) {
  assert(siz[p]);
  siz[p]--;
  int x = head[p];
  head[p] = next[x];
  pre[head[p]] = 0;
  if (x == tail[p]) tail[p] = 0;
  return x;
}
int pop_back(int p) {
  assert(siz[p]);
  siz[p]--;
  int x = tail[p];
  tail[p] = pre[x];
  next[tail[p]] = 0;
  if (x == head[p]) head[p] = 0;
  return x;
}
void maintain() {
  while (siz[0] + 1 < siz[1]) {
    push_back(0, pop_front(1));
  }
  while (siz[0] > siz[1]) {
    push_front(1, pop_back(0));
  }
}
int main() {

const int mx = 2e6 + 10;

bool is[mx];

void init() {
  for (int i = 2; i < mx; i++) {
    if (!is[i]) pri.push_back(i);
    for (int j = 0; j < SZ(pri) && i * pri[j] < mx; j++) {
      is[i * pri[j]] = 1;
      if (i % pri[j] == 0) break;
    }
  }
}
int ans;
mt19937 rng(20021031);
int cnt[mx];
LL b[N];

void work(LL x) {
  for (auto y : pri) {
    if (1ll * y * y > x) break;
    if (x % y) continue;
    do x /= y; while (x % y == 0);
    for (int i = 1; i <= n; i++) {
      ans = max(ans, ++cnt[b[i] = a[i] % y]);
    }
    for (int i = 1; i <= n; i++) --cnt[b[i]];
  }
  if (x != 1) {
    for (int i = 1; i <= n; i++) b[i] = a[i] % x;
    sort(b + 1, b + n + 1);
    for (int i = 1, j; i <= n; i = j) {
      for (j = i; j <= n && b[i] == b[j]; j++) ;
      ans = max(ans, j - i);
    }
  }
}

#include <bits/stdc++.h>
#define LL long long
#define dl double
#define Pi pair<int, int>
#define SZ(a) ((int)a.size())
using namespace std;

const int N = 1e5 + 10;

int T;
int n, s, t, h;
int a[N], b[N];
LL sa[N], sb[N];
LL ans;

void work() {
  LL suma, sumb;
  LL la, ra, lb, rb;

  if (s) {
    suma = sa[n - t] - sa[s - 1], sumb = sb[n - t] - sb[s - 1];
    la = 1, ra = a[s], lb = 1, rb = b[s];
    if (sumb < suma) ans = min(ans, la - rb);
  }

  if (s && t) {
    suma = sa[n - t] - sa[s - 1], sumb = sb[n - t + 1] - sb[s];
    la = 1, ra = a[s], lb = b[n - t + 1], rb = h;
    if (sumb < suma) ans = min(ans, la - rb);
  }

  if (s && t) {
    suma = sa[n - t + 1] - sa[s], sumb = sb[n - t] - sb[s - 1];
    la = a[n - t + 1], ra = h, lb = 1, rb = b[s];
    if (sumb < suma) ans = min(ans, la - rb);
  }

  if (t) {
    suma = sa[n - t + 1] - sa[s], sumb = sb[n - t + 1] - sb[s];
    la = a[n - t + 1], ra = h, lb = b[n - t + 1], rb = h;
    if (sumb < suma) ans = min(ans, la - rb);
  }

  if (s) {
    suma = sa[n - t] - sa[s], sumb = sb[n - t] - sb[s - 1];
    la = a[s], ra = a[n - t + 1], lb = 1, rb = b[s];
    if (sumb - suma < ra) ans = min(ans, max(sumb - suma + 1, la) - rb);
  }

  if (t) {
    suma = sa[n - t] - sa[s], sumb = sb[n - t + 1] - sb[s];
    la = a[s], ra = a[n - t + 1], lb = b[n - t + 1], rb = h;
    if (sumb - suma < ra) ans = min(ans, max(sumb - suma + 1, la) - rb);
  }

  if (s) {
    suma = sa[n - t] - sa[s - 1], sumb = sb[n - t] - sb[s];
    la = 1, ra = a[s], lb = b[s], rb = b[n - t + 1];
    if (sumb + lb < suma) ans = min(ans, la - min(suma - sumb - 1, rb));
  }

  if (t) {
    suma = sa[n - t + 1] - sa[s], sumb = sb[n - t] - sb[s];
    la = a[n - t + 1], ra = h, lb = b[s], rb = b[n - t + 1];
    if (sumb + lb < suma) ans = min(ans, la - min(suma - sumb - 1, rb));
  }

  suma = sa[n - t] - sa[s], sumb = sb[n - t] - sb[s];
  la = a[s], ra = a[n - t + 1], lb = b[s], rb = b[n - t + 1];
  if (sumb - suma < ra - lb) {
    ans = min(ans, max(sumb - suma + 1, la - rb));
  }
}

int main() {
  // freopen("in.txt","r",stdin);
  // freopen("o.txt","w",stdout);
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  cin >> T;
  while (T--) {
    cin >> n >> s >> t >> h;
    std::swap(s, t);
    for (int i = 1; i < n; i++) cin >> a[i];
    for (int i = 1; i < n; i++) cin >> b[i];
    sort(a + 1, a + n);
    sort(b + 1, b + n);
    for (int i = 1; i < n; i++) sa[i] = sa[i - 1] + a[i];
    for (int i = 1; i < n; i++) sb[i] = sb[i - 1] + b[i];
    a[0] = b[0] = 1;
    a[n] = b[n] = h;
    ans = 0x7fffffff;
    n--;
    work();
    if (ans == 0x7fffffff)
      cout << "IMPOSSIBLE\n";
    else
      cout << ans << "\n";
  }
  return 0;
}
