I love tree

树链依次加 \(1^2, 2^2, 3^2, \dots\)，单点询问。\(10^5\)。

轻重链剖分，线段树维护区间 \(1, x, x^2\) 的系数。

// Author:  HolyK
// Created: Thu Jul 22 16:31:11 2021
#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;

constexpr int N(1e5 + 5);
int n;
std::vector<int> g[N];
int dep[N], fa[N], siz[N], son[N];
void dfs(int x) {
  siz[x] = 1;
  for (int y : g[x]) {
    if (y == fa[x]) continue;
    fa[y] = x;
    dep[y] = dep[x] + 1;
    dfs(y);
    siz[x] += siz[y];
    if (siz[y] > siz[son[x]]) son[x] = y;
  }
}
int top[N], in[N];
void dfs2(int x) {
  in[x] = ++in[0];
  if (son[x]) {
    top[son[x]] = top[x];
    dfs2(son[x]);
    for (int y : g[x]) {
      if (y == fa[x] || y == son[x]) continue;
      top[y] = y;
      dfs2(y);
    }
  }
}
int lca(int x, int y) {
  while (top[x] ^ top[y]) {
    if (dep[top[x]] > dep[top[y]]) x = fa[top[x]];
    else y = fa[top[y]];
  }
  return dep[x] < dep[y] ? x : y;
}

std::array<LL, 3> s[N << 2];
#define ls o << 1
#define rs o << 1 | 1
void pushdown(int o, int l, int r) {
  int m = l + r >> 1;
  if (s[o][0] || s[o][1] || s[o][2]) {
    s[ls][0] += s[o][0];
    
    s[ls][1] += s[o][1];
    
    s[ls][2] += s[o][2];

    int len = m - l + 1;
    
    s[rs][0] += s[o][0] + 1LL * len * s[o][1] + 1LL * len * len * s[o][2];
    
    s[rs][1] += s[o][1] + 2LL * len * s[o][2];
    
    s[rs][2] += s[o][2];
    s[o][0] = s[o][1] = s[o][2] = 0;
  }
}
void update(int o, int l, int r, int x, int y, int z) {
  if (x <= l && r <= y) {
    s[o][0] += 1LL * z * z;
    s[o][1] += 2LL * z;
    s[o][2]++;
    return;
  }
  int m = l + r >> 1;
  pushdown(o, l, r);
  if (y <= m) update(ls, l, m, x, y, z);
  else if (x > m) update(rs, m + 1, r, x, y, z);
  else {
    update(ls, l, m, x, m, z);
    update(rs, m + 1, r, m + 1, y, m + 1 - x + z);
  }
}
LL ask(int o, int l, int r, int x) {
  if (l == r) return s[o][0] + s[o][1] + s[o][2];
  int m = l + r >> 1;
  pushdown(o, l, r);
  return x <= m ? ask(ls, l, m, x) : ask(rs, m + 1, r, x);
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  std::cin >> n;
  for (int i = 1, x, y; i < n; i++) {
    std::cin >> x >> y;
    g[x].push_back(y);
    g[y].push_back(x);
  }
  dfs(1);
  top[1] = 1;
  dfs2(1);
  int q;
  std::cin >> q;
  while (q--) {
    int opt, x, y;
    std::cin >> opt >> x;
    if (opt == 1) {
      std::cin >> y;
      int z = lca(x, y);
      int l = 1, r = dep[x] + dep[y] - dep[z] * 2 + 1;
      while (top[x] != top[y]) {
        if (dep[top[x]] > dep[top[y]]) {
          int t = l + dep[x] - dep[top[x]];
          update(1, 1, n, in[top[x]], in[x], -t - 1);
          l = t + 1;
          x = fa[top[x]];
        } else {
          int t = r - dep[y] + dep[top[y]];
          update(1, 1, n, in[top[y]], in[y], t - 1);
          r = t - 1;
          y = fa[top[y]];
        }
        
      }
      if (dep[x] < dep[y]) {
        update(1, 1, n, in[x], in[y], l - 1);
      } else {
        update(1, 1, n, in[y], in[x], -r - 1);
      }
    } else {
      std::cout << ask(1, 1, n, in[x]) << "\n";
    }
  }
  return 0;
}

这复杂度好像还是不太正确，我也不会分析了，反正跑得很快就是了。

using LL = long long;
using PII = std::pair<int, int>;

constexpr int N(1e6 + 5), T(512);
struct Qry {
  int l, r, a, b, id;
  bool operator<(const Qry &rhs) const {
    return l / T == rhs.l / T ? r < rhs.r : l < rhs.l;
  }
} q[N];
int a[N], b[N], sum[N], cnt[N], ans[N];
void add(int p, int x) {
  sum[p / T] += x;
  b[p] += x;
}
int ask(int l, int r) {
  int bl = l / T, br = r / T, ans = 0;
  if (bl == br) {
    for (int i = l; i < r; i++) ans += b[i];
    return ans;
  }
  int y = bl * T + T, x = br * T;
  for (int i = l; i < y; i++) ans += b[i];
  for (int i = x; i < r; i++) ans += b[i];
  for (int i = bl + 1; i < br; i++) ans += sum[i];
  return ans;
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  int n;
  std::cin >> n;
  for (int i = 1; i <= n; i++) {
    std::cin >> a[i];
  }
  int m;
  std::cin >> m;
  for (int i = 1; i <= m; i++) {
    std::cin >> q[i].l >> q[i].r >> q[i].a >> q[i].b;
    q[i].id = i;
  }
  std::sort(q + 1, q + 1 + m);
  int x = 1, y = 0;
  
  for (int i = 1; i <= m; i++) {
    if (q[i].l > q[i].r) continue;
    while (x > q[i].l) {
      x--;
      if (1 == ++cnt[a[x]]) {
        add(a[x], 1);
      }
    }
    while (y < q[i].r) {
      y++;
      if (1 == ++cnt[a[y]]) {
        add(a[y], 1);
      }
    }
    while (y > q[i].r) {
      if (0 == --cnt[a[y]]) {
        add(a[y], -1);
      }
      y--;
    }
    while (x < q[i].l) {
      if (0 == --cnt[a[x]]) {
        add(a[x], -1);
      }
      x++;
    }
    int now = 0;
    for (int j = 22; j >= 0; j--) {
      if (q[i].b >> j & 1) {
        int l = now;
        if (q[i].a >> j & 1) {
          l += 1 << j;
        } 
        ans[q[i].id] += ask(l, l + (1 << j));
      }
      if ((q[i].a ^ q[i].b) >> j & 1) {
        now |= 1 << j;
      }
    }
    ans[q[i].id] += b[q[i].a ^ q[i].b];
  }
  for (int i = 1; i <= m; i++) {
    std::cout << ans[i] << "\n";
  }
  return 0;
}

    }
  }
  return c;
}
#define ls o << 1
#define rs o << 1 | 1
void pushup(int o) {
  sum[o] = sum[ls] + sum[rs];
}
void inc(int o, int l, int r, int z, int t) {
  inc(add[o][t], z);
  int s = (r - l + 1LL) * z % P;
  sum[o][t + 2] = (sum[o][t + 2] + 2LL * sum[o][t] * z + 1LL * s * z) % P;
  sum[o][4] = (sum[o][4] + 1LL * sum[o][t ^ 1] * z) % P;
  inc(sum[o][t], s);
}
void mul(int o, const Matrix &x) {
  add[o] = {
    (int)((1LL * x[0][0] * add[o][0] + 1LL * x[0][1] * add[o][1]) % P),
    (int)((1LL * x[1][0] * add[o][0] + 1LL * x[1][1] * add[o][1]) % P)
  };
  tag[o] = x * tag[o];
  sum[o] = x * sum[o];
}

void pushdown(int o, int l, int r) {
  int m = l + r >> 1;
  if (tag[o] != One) {
    mul(ls, tag[o]);
    mul(rs, tag[o]);
    tag[o] = One;
  }
  for (int t = 0; t < 2; t++) {
    if (add[o][t]) {
      inc(ls, l, m, add[o][t], t);
      inc(rs, m + 1, r, add[o][t], t);
      add[o][t] = 0;
    }
  }
}
void update(int o, int l, int r, int x, int y, int z, int t) {
  if (x <= l && r <= y) {
    inc(o, l, r, z, t);
    return;
  }
  int m = l + r >> 1;
  pushdown(o, l, r);
  if (x <= m) update(ls, l, m, x, y, z, t);
  if (y > m) update(rs, m + 1, r, x, y, z, t);
  pushup(o);
}
void update(int o, int l, int r, int x, int y, const Matrix &z) {

inline int sumMod(int x, int y) {

}

  int r = 1;

    if (k & 1) r = 1LL * r * x % P;
  }
  return r;
}
constexpr int N(2e5 + 5);
typedef std::array<int, 5> Data;
typedef std::array<int, 2> Vector;
typedef std::array<std::array<int, 2>, 2> Matrix;
const Matrix One = {

  Vector({0, 1})
}, Rev = {
  Vector({0, 1}),
  Vector({1, 0})
}, Bt = {
  Vector({3, 2}),
  Vector({3, P - 2})

Data sum[N << 2];
Vector add[N << 2];
Matrix tag[N << 2];
Data operator+(Data a, Data b) {
  for (int i = 0; i < 5; i++) {
    inc(a[i], b[i]);
  }
  return a;
}
inline int mod(LL x) { return x % P; }
Data operator*(Matrix a, Data b) {
  return {
    mod(1LL * a[0][0] * b[0] + 1LL * a[0][1] * b[1]),
    mod(1LL * a[1][0] * b[0] + 1LL * a[1][1] * b[1]),
    mod(1LL * a[0][0] * a[0][0] % P * b[2] + 1LL * a[0][1] * a[0][1] % P * b[3] + 2LL * a[0][0] * a[0][1] % P * b[4]),
    mod(1LL * a[1][0] * a[1][0] % P * b[2] + 1LL * a[1][1] * a[1][1] % P * b[3] + 2LL * a[1][0] * a[1][1] % P * b[4]),
    mod(1LL * a[0][0] * a[1][0] % P * b[2] + 1LL * a[0][1] * a[1][1] % P * b[3] + (1LL * a[0][0] * a[1][1] + 1LL * a[0][1] * a[1][0]) % P * b[4])
  };
}
Matrix operator*(Matrix a, Matrix b) {
  Matrix c;
  for (int i = 0; i < 2; i++) {
    for (int j = 0; j < 2; j++) {
      c[i][j] = (1LL * a[i][0] * b[0][j] + 1LL * a[i][1] * b[1][j]) % P;
    }
  }
  return c;
}

  inc(sum[o][t], s);
}

  };

}

void pushdown(int o, int l, int r) {

  if (tag[o] != One) {
    mul(ls, tag[o]);
    mul(rs, tag[o]);
    tag[o] = One;
  }
  for (int t = 0; t < 2; t++) {
    if (add[o][t]) {
      inc(ls, l, m, add[o][t], t);
      inc(rs, m + 1, r, add[o][t], t);
      add[o][t] = 0;
    }
  }
}
void update(int o, int l, int r, int x, int y, int z, int t) {
  if (x <= l && r <= y) {
    inc(o, l, r, z, t);
    return;
  }
  int m = l + r >> 1;
  pushdown(o, l, r);
  if (x <= m) update(ls, l, m, x, y, z, t);
  if (y > m) update(rs, m + 1, r, x, y, z, t);
  pushup(o);
}
void update(int o, int l, int r, int x, int y, const Matrix &z) {
  if (x <= l && r <= y) {
    mul(o, z);
    return;
  }

  if (x <= m) update(ls, l, m, x, y, z);
  if (y > m) update(rs, m + 1, r, x, y, z);
  pushup(o);
}
inline int sumMod(int x, int y) {
  return x + y >= P ? x + y - P : x + y;
}
int ask(int o, int l, int r, int x, int y) {
  if (r < x || y < l) return 0;
  if (x <= l && r <= y) return sum[o][4];
  int m = l + r >> 1;
  pushdown(o, l, r);
  return sumMod(ask(ls, l, m, x, y), ask(rs, m + 1, r, x, y));
}

}

质因数分解。

  1. 只含有小于 \(\sqrt{V}(V = 100000)\) 的质因数的三元组，数量不多，先搜索存下来，然后直接计算。

  2. 如果含有大于 \(\sqrt{V}\) 的质因数，一定是三元组中的某两个数含有这个质因数。

    枚举这个质因数，然后平方枚举剩下的因数。

    这里复杂度是 \(O\left(\sum\left(\dfrac{V}{p}\right)^2\right)\)，放缩再积分一下得到 \(O(V \sqrt{V})\)，实际表现更好。

// Author:  HolyK
// Created: Fri Jul 23 15:47:33 2021
#include <bits/stdc++.h>
#define dbg(a...) fprintf(stderr, a)
template <class T, class U>
inline bool smin(T &x, const U &y) {
  return y < x ? x = y, 1 : 0;
}
template <class T, class U>
inline bool smax(T &x, const U &y) {
  return x < y ? x = y, 1 : 0;
}

using LL = long long;
using PII = std::pair<int, int>;

constexpr int N(1e5 + 5), M(1e5), T(350);
bool np[N];
std::vector<int> primes;
void sieve(int n) {
  for (int i = 2; i <= n; i++) {
    if (!np[i]) primes.push_back(i);
    for (int j : primes) {
      if (1LL * i * j > n) break;
      np[i * j] = true;
      if (i % j == 0) break;
    }
  }
}
int id[N], cnt[N], val[T][T];
std::vector<int> g[N];
std::vector<std::array<int, 3>> small;
LL cal(int a, int b, int c) {
  if (a == c) {
    return 1LL * cnt[a] * (cnt[a] - 1) * (cnt[a] - 2) / 6;
  }
  if (a == b || b == c) {
    return 1LL * cnt[b] * (cnt[b] - 1) / 2 * cnt[a ^ b ^ c];
  }
  return 1LL * cnt[a] * cnt[b] * cnt[c];
}
void solve() {
  memset(id, 0, sizeof id);
  memset(cnt, 0, sizeof cnt);
  int n, big = 0;
  std::cin >> n;
  while (n--) {
    int x, y = 1;
    std::cin >> x;
    for (int i : primes) {
      if (x % i) continue;
      int c = 0;
      do {
        x /= i;
        c ^= 1;
      } while (x % i == 0);
      if (c & 1) y *= i;
    }
    if (x > 1) {
      if (!id[x]) id[x] = ++big, g[big].clear();
      g[id[x]].push_back(y);
    } else {
      cnt[y]++;
    }
  }
  LL ans = 0;
  for (auto &a : small) ans += cal(a[0], a[1], a[2]);
  for (int i = 1; i <= big; i++) {
    std::vector<int> c(1 + *std::max_element(g[i].begin(), g[i].end()));
    for (int x : g[i]) c[x]++;
    for (int j = 1; j < c.size(); j++) {
      if (!c[j]) continue;
      ans += 1LL * cnt[1] * c[j] * (c[j] - 1) / 2;
      for (int k = j + 1; k < c.size(); k++) {
        if (!c[k]) continue;
        ans += 1LL * c[j] * c[k] * cnt[val[j][k]];
      }
    }
  }
  std::cout << ans << "\n";
}
void dfs(int p, int a, int b, int c) {
  if (a > M || b > M || c > M) return;
  if (p == -1) {
    if (a <= b && b <= c) {
      small.push_back({a, b, c});
    }
    return;
  }
  dfs(p - 1, a, b, c);
  int x = primes[p];
  dfs(p - 1, a * x, b, c * x);
  dfs(p - 1, a, b * x, c * x);
  dfs(p - 1, a * x, b * x, c);
}
int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  sieve(sqrt(M));
  small.reserve(1584455);
  dfs(primes.size() - 1, 1, 1, 1);
  for (int i = 1; i < T; i++) {
    for (int j = 1; j < T; j++) {
      int x = std::__gcd(i, j);
      val[i][j] = i / x * j / x;
    }
  }
  int t;
  std::cin >> t;
  while (t--) { 
    solve();
  }
  return 0;
}
